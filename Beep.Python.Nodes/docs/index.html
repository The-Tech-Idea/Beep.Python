?? Overview

Beep.Python.Nodes provides a comprehensive visual node-based system for Python integration, enabling users to create complex Python workflows through an intuitive drag-and-drop interface. The system supports visual programming, data flow management, and enterprise-grade node orchestration.

Key Features
  • Visual Programming: Drag-and-drop interface for creating Python workflows
  • Node Libraries: Comprehensive collection of pre-built Python integration nodes
  • Custom Nodes: Create and deploy custom nodes for specific use cases
  • Data Flow Management: Visual data flow with type checking and validation
  • Runtime Integration: Seamless integration with Python runtime environments
  • Enterprise Features: Node versioning, deployment, and management capabilities
  • AI Integration: Specialized nodes for AI and machine learning workflows

PythonRuntimeNode

Core node responsible for Python runtime management, environment initialization, and execution orchestration.

Runtime Node Configuration

public class PythonRuntimeNode : IWorkFlowStepNode
{
    // Node Properties
    public string NodeName { get; set; } = "Python Runtime";
    public string NodeType { get; set; } = "Runtime";
    public string NodeCategory { get; set; } = "Python";
    
    // Runtime Configuration
    public string PythonPath { get; set; }
    public string VirtualEnvironmentPath { get; set; }
    public Dictionary<string, object> EnvironmentVariables { get; set; }
    
    // Execution Properties
    public bool IsInitialized { get; private set; }
    public RuntimeStatus Status { get; private set; }
    public TimeSpan ExecutionTimeout { get; set; } = TimeSpan.FromMinutes(30);
    
    // Input/Output Ports
    public List<NodePort> InputPorts { get; set; }
    public List<NodePort> OutputPorts { get; set; }
    
    // Node Methods
    public async Task<NodeExecutionResult> ExecuteAsync(NodeExecutionContext context)
    {
        try
        {
            // Initialize Python runtime if needed
            if (!IsInitialized)
            {
                await InitializeRuntimeAsync();
            }
            
            // Execute Python code or workflow
            var result = await ExecutePythonWorkflowAsync(context);
            
            return new NodeExecutionResult
            {
                Success = true,
                OutputData = result,
                ExecutionTime = DateTime.UtcNow,
                Message = "Python runtime executed successfully"
            };
        }
        catch (Exception ex)
        {
            return new NodeExecutionResult
            {
                Success = false,
                Error = ex,
                Message = $"Runtime execution failed: {ex.Message}"
            };
        }
    }
    
    private async Task InitializeRuntimeAsync()
    {
        // Initialize Python.NET
        var runtimeManager = ServiceLocator.GetService<IPythonRuntimeManager>();
        
        // Configure environment
        if (!string.IsNullOrEmpty(VirtualEnvironmentPath))
        {
            await runtimeManager.SetVirtualEnvironmentAsync(VirtualEnvironmentPath);
        }
        
        // Set environment variables
        foreach (var envVar in EnvironmentVariables)
        {
            Environment.SetEnvironmentVariable(envVar.Key, envVar.Value.ToString());
        }
        
        // Initialize runtime
        IsInitialized = await runtimeManager.InitializeAsync();
        Status = IsInitialized ? RuntimeStatus.Ready : RuntimeStatus.Error;
    }
}

PythonVirtualEnvNode

Specialized node for virtual environment management, package installation, and environment isolation.

Virtual Environment Node

public class PythonVirtualEnvNode : IWorkFlowStepNode
{
    // Node Configuration
    public string NodeName { get; set; } = "Virtual Environment";
    public string EnvironmentName { get; set; }
    public string BasePythonPath { get; set; }
    public string EnvironmentPath { get; set; }
    
    // Package Management
    public List<string> RequiredPackages { get; set; } = new();
    public string RequirementsFile { get; set; }
    public bool AutoInstallPackages { get; set; } = true;
    public bool UpgradePackages { get; set; } = false;
    
    // Environment Operations
    public EnvironmentOperation Operation { get; set; }
    
    public async Task<NodeExecutionResult> ExecuteAsync(NodeExecutionContext context)
    {
        var envManager = ServiceLocator.GetService<IPythonVirtualEnvManager>();
        
        try
        {
            switch (Operation)
            {
                case EnvironmentOperation.Create:
                    return await CreateEnvironmentAsync(envManager);
                    
                case EnvironmentOperation.Activate:
                    return await ActivateEnvironmentAsync(envManager);
                    
                case EnvironmentOperation.InstallPackages:
                    return await InstallPackagesAsync(envManager);
                    
                case EnvironmentOperation.UpdatePackages:
                    return await UpdatePackagesAsync(envManager);
                    
                case EnvironmentOperation.Delete:
                    return await DeleteEnvironmentAsync(envManager);
                    
                default:
                    throw new InvalidOperationException($"Unknown operation: {Operation}");
            }
        }
        catch (Exception ex)
        {
            return new NodeExecutionResult
            {
                Success = false,
                Error = ex,
                Message = $"Virtual environment operation failed: {ex.Message}"
            };
        }
    }
    
    private async Task<NodeExecutionResult> CreateEnvironmentAsync(IPythonVirtualEnvManager envManager)
    {
        var success = await envManager.CreateVirtualEnvironmentAsync(
            EnvironmentName, 
            BasePythonPath, 
            EnvironmentPath);
            
        if (success && AutoInstallPackages)
        {
            await InstallRequiredPackagesAsync(envManager);
        }
        
        return new NodeExecutionResult
        {
            Success = success,
            OutputData = new { EnvironmentPath = EnvironmentPath },
            Message = success ? "Virtual environment created successfully" : "Failed to create virtual environment"
        };
    }
    
    private async Task InstallRequiredPackagesAsync(IPythonVirtualEnvManager envManager)
    {
        // Install from requirements file
        if (!string.IsNullOrEmpty(RequirementsFile) && File.Exists(RequirementsFile))
        {
            await envManager.InstallFromRequirementsAsync(RequirementsFile);
        }
        
        // Install individual packages
        foreach (var package in RequiredPackages)
        {
            await envManager.InstallPackageAsync(package, upgradeIfExists: UpgradePackages);
        }
    }
}

public enum EnvironmentOperation
{
    Create,
    Activate,
    InstallPackages,
    UpdatePackages,
    Delete
}

AICPythonNode

Advanced AI-focused node for machine learning workflows, model operations, and intelligent data processing.

AI Python Node

public class AICPythonNode : IWorkFlowStepNode
{
    // AI Configuration
    public string NodeName { get; set; } = "AI Python Node";
    public AIOperation Operation { get; set; }
    public string ModelType { get; set; }
    public string ModelPath { get; set; }
    
    // ML Framework Configuration
    public MLFramework Framework { get; set; } = MLFramework.ScikitLearn;
    public Dictionary<string, object> ModelParameters { get; set; } = new();
    
    // Data Configuration
    public string InputDataPath { get; set; }
    public string OutputDataPath { get; set; }
    public string TargetColumn { get; set; }
    public List<string> FeatureColumns { get; set; } = new();
    
    // Training Configuration
    public double TestSize { get; set; } = 0.2;
    public int RandomState { get; set; } = 42;
    public bool CrossValidate { get; set; } = true;
    public int CVFolds { get; set; } = 5;
    
    public async Task<NodeExecutionResult> ExecuteAsync(NodeExecutionContext context)
    {
        var mlManager = ServiceLocator.GetService<IPythonMLManager>();
        
        try
        {
            switch (Operation)
            {
                case AIOperation.TrainModel:
                    return await TrainModelAsync(mlManager, context);
                    
                case AIOperation.PredictModel:
                    return await PredictAsync(mlManager, context);
                    
                case AIOperation.EvaluateModel:
                    return await EvaluateModelAsync(mlManager, context);
                    
                case AIOperation.PreprocessData:
                    return await PreprocessDataAsync(mlManager, context);
                    
                case AIOperation.FeatureEngineering:
                    return await FeatureEngineeringAsync(mlManager, context);
                    
                default:
                    throw new InvalidOperationException($"Unknown AI operation: {Operation}");
            }
        }
        catch (Exception ex)
        {
            return new NodeExecutionResult
            {
                Success = false,
                Error = ex,
                Message = $"AI operation failed: {ex.Message}"
            };
        }
    }
    
    private async Task<NodeExecutionResult> TrainModelAsync(IPythonMLManager mlManager, NodeExecutionContext context)
    {
        // Load data
        await mlManager.LoadDataAsync(InputDataPath);
        
        // Prepare data
        await mlManager.PrepareDataAsync(new DataPreparationOptions
        {
            TargetColumn = TargetColumn,
            FeatureColumns = FeatureColumns,
            TestSize = TestSize,
            RandomState = RandomState
        });
        
        // Configure algorithm
        var algorithm = mlManager.GetAlgorithm(ModelType);
        algorithm.SetParameters(ModelParameters);
        
        // Train model
        var trainingResult = await mlManager.TrainModelAsync(algorithm);
        
        // Cross-validate if requested
        CrossValidationResult cvResult = null;
        if (CrossValidate)
        {
            cvResult = await mlManager.CrossValidateAsync(algorithm, CVFolds);
        }
        
        // Save model
        if (!string.IsNullOrEmpty(ModelPath))
        {
            await mlManager.SaveModelAsync(algorithm.Name, ModelPath);
        }
        
        return new NodeExecutionResult
        {
            Success = trainingResult.Success,
            OutputData = new
            {
                TrainingScore = trainingResult.TrainingScore,
                ValidationScore = trainingResult.ValidationScore,
                TrainingTime = trainingResult.TrainingTime,
                CrossValidationScore = cvResult?.MeanScore,
                ModelPath = ModelPath
            },
            Message = $"Model training completed. Training score: {trainingResult.TrainingScore:F4}"
        };
    }
    
    private async Task<NodeExecutionResult> PredictAsync(IPythonMLManager mlManager, NodeExecutionContext context)
    {
        // Load model
        if (!string.IsNullOrEmpty(ModelPath))
        {
            await mlManager.LoadModelAsync(ModelType, ModelPath);
        }
        
        // Load input data
        var inputData = context.GetInputData<double[][]>("InputData");
        if (inputData == null && !string.IsNullOrEmpty(InputDataPath))
        {
            await mlManager.LoadDataAsync(InputDataPath);
            inputData = await mlManager.GetDataForPredictionAsync();
        }
        
        // Make predictions
        var predictions = await mlManager.PredictAsync(inputData);
        
        // Save predictions if output path specified
        if (!string.IsNullOrEmpty(OutputDataPath))
        {
            await mlManager.SavePredictionsAsync(predictions, OutputDataPath);
        }
        
        return new NodeExecutionResult
        {
            Success = true,
            OutputData = new
            {
                Predictions = predictions,
                PredictionCount = predictions.Length,
                OutputPath = OutputDataPath
            },
            Message = $"Generated {predictions.Length} predictions"
        };
    }
}

    FeatureEngineering,
    HyperparameterTuning
}

Node-Based Workflow Creation

public class PythonWorkflowDesigner
{
    private readonly NodeRegistry _nodeRegistry;
    private readonly WorkflowEngine _workflowEngine;
    
    public PythonWorkflowDesigner()
    {
        _nodeRegistry = new NodeRegistry();
        _workflowEngine = new WorkflowEngine();
        
        // Register built-in nodes
        RegisterBuiltInNodes();
    }
    
    private void RegisterBuiltInNodes()
    {
        _nodeRegistry.RegisterNode<PythonRuntimeNode>("Python Runtime", "Runtime");
        _nodeRegistry.RegisterNode<PythonVirtualEnvNode>("Virtual Environment", "Environment");
        _nodeRegistry.RegisterNode<AICPythonNode>("AI Python", "AI/ML");
        _nodeRegistry.RegisterNode<DataLoaderNode>("Data Loader", "Data");
        _nodeRegistry.RegisterNode<DataProcessorNode>("Data Processor", "Data");
        _nodeRegistry.RegisterNode<ModelTrainerNode>("Model Trainer", "AI/ML");
        _nodeRegistry.RegisterNode<PredictionNode>("Prediction", "AI/ML");
        _nodeRegistry.RegisterNode<OutputNode>("Output", "Output");
    }
    
    public WorkflowDefinition CreateMLWorkflow()
    {
        var workflow = new WorkflowDefinition("ML Pipeline");
        
        // Create nodes
        var envNode = _nodeRegistry.CreateNode<PythonVirtualEnvNode>("setup-env");
        envNode.EnvironmentName = "ml-pipeline";
        envNode.RequiredPackages = new List<string> { "pandas", "scikit-learn", "numpy" };
        envNode.Operation = EnvironmentOperation.Create;
        
        var dataLoader = _nodeRegistry.CreateNode<DataLoaderNode>("load-data");
        dataLoader.DataSource = "data/training_data.csv";
        dataLoader.DataFormat = DataFormat.CSV;
        
        var dataProcessor = _nodeRegistry.CreateNode<DataProcessorNode>("process-data");
        dataProcessor.Operations = new[]
        {
            DataOperation.HandleMissingValues,
            DataOperation.ScaleFeatures,
            DataOperation.EncodeCategorical
        };
        
        var trainer = _nodeRegistry.CreateNode<AICPythonNode>("train-model");
        trainer.Operation = AIOperation.TrainModel;
        trainer.ModelType = "RandomForestClassifier";
        trainer.Framework = MLFramework.ScikitLearn;
        trainer.CrossValidate = true;
        
        var evaluator = _nodeRegistry.CreateNode<AICPythonNode>("evaluate-model");
        evaluator.Operation = AIOperation.EvaluateModel;
        
        var output = _nodeRegistry.CreateNode<OutputNode>("save-results");
        output.OutputPath = "results/model_results.json";
        
        // Add nodes to workflow
        workflow.AddNode(envNode);
        workflow.AddNode(dataLoader);
        workflow.AddNode(dataProcessor);
        workflow.AddNode(trainer);
        workflow.AddNode(evaluator);
        workflow.AddNode(output);
        
        // Create connections
        workflow.ConnectNodes(envNode, "EnvironmentReady", dataLoader, "Environment");
        workflow.ConnectNodes(dataLoader, "Data", dataProcessor, "InputData");
        workflow.ConnectNodes(dataProcessor, "ProcessedData", trainer, "TrainingData");
        workflow.ConnectNodes(trainer, "TrainedModel", evaluator, "Model");
        workflow.ConnectNodes(evaluator, "EvaluationResults", output, "Data");
        
        return workflow;
    }
    
    public async Task<WorkflowExecutionResult> ExecuteWorkflowAsync(WorkflowDefinition workflow)
    {
        return await _workflowEngine.ExecuteAsync(workflow);
    }
}

public class DataVisualizationNode : IWorkFlowStepNode
{
    public string NodeName { get; set; } = "Data Visualization";
    public string NodeType { get; set; } = "Visualization";
    public string NodeCategory { get; set; } = "Data";
    
    // Configuration
    public VisualizationType ChartType { get; set; }
    public string XColumn { get; set; }
    public string YColumn { get; set; }
    public string ColorColumn { get; set; }
    public string OutputPath { get; set; }
    
    // Visualization settings
    public ChartSettings Settings { get; set; } = new();
    
    public async Task<NodeExecutionResult> ExecuteAsync(NodeExecutionContext context)
    {
        var plotManager = ServiceLocator.GetService<IPythonPlotManager>();
        
        try
        {
            // Get input data
            var dataFrame = context.GetInputData<string>("DataFrame");
            
            // Create visualization based on type
            string plotCode = GeneratePlotCode();
            
            // Execute visualization
            await plotManager.ExecutePlotCodeAsync(plotCode);
            
            // Save plot
            if (!string.IsNullOrEmpty(OutputPath))
            {
                await plotManager.SavePlotAsync(OutputPath);
            }
            
            return new NodeExecutionResult
            {
                Success = true,
                OutputData = new { PlotPath = OutputPath },
                Message = $"Visualization created: {ChartType}"
            };
        }
        catch (Exception ex)
        {
            return new NodeExecutionResult
            {
                Success = false,
                Error = ex,
                Message = $"Visualization failed: {ex.Message}"
            };
        }
    }
    
    private string GeneratePlotCode()
    {
        return ChartType switch
        {
            VisualizationType.Scatter => GenerateScatterPlot(),
            VisualizationType.Line => GenerateLinePlot(),
            VisualizationType.Bar => GenerateBarPlot(),
            VisualizationType.Histogram => GenerateHistogram(),
            VisualizationType.Heatmap => GenerateHeatmap(),
            _ => throw new NotSupportedException($"Chart type {ChartType} not supported")
        };
    }
}

Node Security Configuration

public class NodeSecurityManager
{
    public async Task<bool> ValidateNodeSecurityAsync(IWorkFlowStepNode node, NodeExecutionContext context)
    {
        // Check node permissions
        if (!await CheckNodePermissionsAsync(node, context.User))
        {
            throw new UnauthorizedAccessException($"User {context.User} not authorized to execute node {node.NodeName}");
        }
        
        // Validate node configuration
        var validationResult = await ValidateNodeConfigurationAsync(node);
        if (!validationResult.IsValid)
        {
            throw new InvalidOperationException($"Node configuration invalid: {string.Join(", ", validationResult.Errors)}");
        }
        
        // Check resource limits
        if (!await CheckResourceLimitsAsync(node, context))
        {
            throw new InvalidOperationException("Node execution would exceed resource limits");
        }
        
        // Audit log
        await LogNodeExecutionAsync(node, context, "AUTHORIZED");
        
        return true;
    }
    
    private async Task<bool> CheckNodePermissionsAsync(IWorkFlowStepNode node, string user)
    {
        var permissions = await GetUserPermissionsAsync(user);
        return permissions.CanExecuteNodeType(node.NodeType) && 
               permissions.CanAccessNodeCategory(node.NodeCategory);
    }
}

Performance Monitoring

Comprehensive monitoring of node execution, resource usage, and performance metrics.

Node Performance Monitoring

public class NodePerformanceMonitor
{
    private readonly IMetricsCollector _metricsCollector;
    
    public async Task<NodeExecutionResult> MonitorNodeExecutionAsync(
        IWorkFlowStepNode node, 
        NodeExecutionContext context)
    {
        var stopwatch = Stopwatch.StartNew();
        var startMemory = GC.GetTotalMemory(false);
        
        try
        {
            // Record execution start
            await _metricsCollector.RecordNodeStartAsync(node.NodeName, context.WorkflowId);
            
            // Execute node
            var result = await node.ExecuteAsync(context);
            
            stopwatch.Stop();
            var endMemory = GC.GetTotalMemory(false);
            
            // Record performance metrics
            var metrics = new NodeExecutionMetrics
            {
                NodeName = node.NodeName,
                NodeType = node.NodeType,
                ExecutionTime = stopwatch.Elapsed,
                MemoryUsed = endMemory - startMemory,
                Success = result.Success,
                Timestamp = DateTime.UtcNow
            };
            
            await _metricsCollector.RecordNodeMetricsAsync(metrics);
            
            return result;
        }
        catch (Exception ex)
        {
            stopwatch.Stop();
            
            // Record error metrics
            await _metricsCollector.RecordNodeErrorAsync(node.NodeName, ex, stopwatch.Elapsed);
            
            throw;
        }
    }
}

? Best Practices

Node Development Best Practices
  • Error Handling: Always implement comprehensive error handling in nodes
  • Resource Management: Properly dispose of resources and clean up after execution
  • Configuration Validation: Validate all node configuration before execution
  • Progress Reporting: Provide progress updates for long-running operations
  • Logging: Implement detailed logging for debugging and monitoring
  • Testing: Create comprehensive unit tests for custom nodes
Common Pitfalls
  • Memory Leaks: Ensure proper cleanup of Python objects and resources
  • Thread Safety: Consider thread safety when accessing shared resources
  • Exception Propagation: Handle exceptions gracefully without breaking the workflow
  • Data Validation: Always validate input data before processing

public async Task CreateMLPipelineWorkflow()
{
    var designer = new PythonWorkflowDesigner();
    
    // Create comprehensive ML pipeline
    var workflow = new WorkflowDefinition("Customer Churn Prediction Pipeline");
    
    // Environment setup
    var envSetup = new PythonVirtualEnvNode
    {
        NodeName = "ML Environment",
        EnvironmentName = "churn-prediction",
        RequiredPackages = new List<string>
        {
            "pandas>=1.5.0",
            "scikit-learn>=1.2.0",
            "numpy>=1.21.0",
            "matplotlib>=3.6.0",
            "seaborn>=0.12.0"
        },
        Operation = EnvironmentOperation.Create
    };
    
    // Data loading
    var dataLoader = new DataLoaderNode
    {
        NodeName = "Load Customer Data",
        DataSource = "data/customer_data.csv",
        DataFormat = DataFormat.CSV
    };
    
    // Data preprocessing
    var preprocessor = new DataProcessorNode
    {
        NodeName = "Preprocess Data",
        Operations = new[]
        {
            DataOperation.HandleMissingValues,
            DataOperation.EncodeCategorical,
            DataOperation.ScaleFeatures,
            DataOperation.FeatureSelection
        }
    };
    
    // Model training
    var trainer = new AICPythonNode
    {
        NodeName = "Train Churn Model",
        Operation = AIOperation.TrainModel,
        ModelType = "RandomForestClassifier",
        Framework = MLFramework.ScikitLearn,
        TargetColumn = "churn",
        CrossValidate = true,
        CVFolds = 5,
        ModelParameters = new Dictionary<string, object>
        {
            ["n_estimators"] = 100,
            ["max_depth"] = 10,
            ["random_state"] = 42
        }
    };
    
    // Model evaluation
    var evaluator = new AICPythonNode
    {
        NodeName = "Evaluate Model",
        Operation = AIOperation.EvaluateModel
    };
    
    // Visualization
    var visualizer = new DataVisualizationNode
    {
        NodeName = "Generate Reports",
        ChartType = VisualizationType.Heatmap,
        OutputPath = "reports/feature_importance.png"
    };
    
    // Build workflow
    workflow.AddNode(envSetup);
    workflow.AddNode(dataLoader);
    workflow.AddNode(preprocessor);
    workflow.AddNode(trainer);
    workflow.AddNode(evaluator);
    workflow.AddNode(visualizer);
    
    // Connect nodes
    workflow.ConnectNodes(envSetup, "Ready", dataLoader, "Environment");
    workflow.ConnectNodes(dataLoader, "Data", preprocessor, "InputData");
    workflow.ConnectNodes(preprocessor, "ProcessedData", trainer, "TrainingData");
    workflow.ConnectNodes(trainer, "Model", evaluator, "TrainedModel");
    workflow.ConnectNodes(evaluator, "Results", visualizer, "Data");
    
    // Execute workflow
    var result = await designer.ExecuteWorkflowAsync(workflow);
    
    if (result.Success)
    {
        Console.WriteLine("ML Pipeline completed successfully!");
        Console.WriteLine($"Model accuracy: {result.GetOutput<double>("Accuracy"):F4}");
    }
    else
    {
        Console.WriteLine($"Pipeline failed: {result.ErrorMessage}");
    }
}
