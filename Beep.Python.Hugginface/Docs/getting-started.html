Prerequisites

Before you start using Beep.Python.AI.Transformers, ensure you have the following:

System Requirements
  • .NET 6, 7, 8, or 9 - Any supported .NET version
  • Python 3.8+ installed on your system
  • Python.NET for Python interoperability
  • Virtual Environment Support (recommended)
  • API Keys for your preferred AI providers
Supported Platforms
  • Windows 10/11
  • macOS 10.15+
  • Linux (Ubuntu 18.04+)
  • Docker containers

Basic Setup

1. Configure Services (ASP.NET Core)

using Beep.Python.AI.Transformers;
using Beep.Python.Model;

// In Program.cs or Startup.cs
builder.Services.AddSingleton<IPythonRunTimeManager, PythonNetRunTimeManager>();
builder.Services.AddSingleton<IPythonCodeExecuteManager, PythonCodeExecuteManager>();

// Configure AI provider connections
builder.Services.Configure<OpenAIConnectionConfig>(
    builder.Configuration.GetSection("OpenAI"));
builder.Services.Configure<AzureOpenAIConnectionConfig>(
    builder.Configuration.GetSection("AzureOpenAI"));

2. Configuration (appsettings.json)

{
  "OpenAI": {
    "ApiKey": "sk-your-openai-api-key",
    "TimeoutSeconds": 60,
    "MaxRetries": 3
  },
  "AzureOpenAI": {
    "Endpoint": "https://your-resource.openai.azure.com/",
    "ApiKey": "your-azure-api-key",
    "ApiVersion": "2024-02-01",
    "DeploymentName": "gpt-35-turbo"
  }
}

3. Basic Usage Example

using Beep.Python.AI.Transformers;

public class AIService
{
    private readonly IPythonRunTimeManager _runtimeManager;
    private readonly IPythonCodeExecuteManager _executeManager;

    public AIService(
        IPythonRunTimeManager runtimeManager,
        IPythonCodeExecuteManager executeManager)
    {
        _runtimeManager = runtimeManager;
        _executeManager = executeManager;
    }

    public async Task<string> GenerateTextAsync(string prompt)
    {
        // Configure OpenAI connection
        var openAIConfig = new OpenAIConnectionConfig
        {
            ApiKey = "sk-your-api-key",
            TimeoutSeconds = 60
        };

        // Create pipeline
        var pipeline = TransformerPipelineFactory.CreateOpenAIPipeline(
            _runtimeManager,
            _executeManager,
            openAIConfig
        );

        // Initialize and load model
        await pipeline.InitializeAsync(new TransformerPipelineConfig
        {
            TaskType = TransformerTask.TextGeneration
        });

        await pipeline.LoadModelAsync(new TransformerModelInfo
        {
            Name = "gpt-4",
            Source = TransformerModelSource.OpenAI
        }, TransformerTask.TextGeneration);

        // Generate text
        var result = await pipeline.GenerateTextAsync(prompt);
        return result.Success ? result.Data : $"Error: {result.ErrorMessage}";
    }
}

Multiple Providers Example

public class MultiProviderService
{
    private readonly Dictionary<TransformerModelSource, ITransformerPipeLine> _pipelines = new();

    public async Task InitializeProvidersAsync()
    {
        // OpenAI
        _pipelines[TransformerModelSource.OpenAI] = 
            TransformerPipelineFactory.CreateOpenAIPipeline(
                runtimeManager, executeManager, openAIConfig);

        // Azure
        _pipelines[TransformerModelSource.Azure] = 
            TransformerPipelineFactory.CreateAzureOpenAIPipeline(
                runtimeManager, executeManager, azureConfig);

        // Anthropic
        _pipelines[TransformerModelSource.Anthropic] = 
            TransformerPipelineFactory.CreateAnthropicPipeline(
                runtimeManager, executeManager, anthropicConfig);

        // Initialize all pipelines
        foreach (var pipeline in _pipelines.Values)
        {
            await pipeline.InitializeAsync(new TransformerPipelineConfig
            {
                TaskType = TransformerTask.TextGeneration
            });
        }
    }

    public async Task<string> GenerateWithBestProviderAsync(string prompt)
    {
        // Try providers in order of preference
        var providers = new[] { 
            TransformerModelSource.OpenAI, 
            TransformerModelSource.Azure, 
            TransformerModelSource.Anthropic 
        };

        foreach (var provider in providers)
        {
            try
            {
                var result = await _pipelines[provider].GenerateTextAsync(prompt);
                if (result.Success)
                    return result.Data;
            }
            catch (Exception ex)
            {
                // Log error and try next provider
                Console.WriteLine($"Provider {provider} failed: {ex.Message}");
            }
        }

        throw new Exception("All providers failed");
    }
}

Enterprise Setup

Session Management for Multi-User Environments

public class EnterpriseAIService
{
    private readonly IPythonRunTimeManager _runtimeManager;
    private readonly IPythonCodeExecuteManager _executeManager;

    public async Task<ITransformerPipeLine> CreateUserPipelineAsync(
        string userId, 
        TransformerModelSource provider)
    {
        // Create session-aware pipeline with isolation
        var pipeline = TransformerPipelineFactory.CreateSessionAwarePipelineWithConnection(
            provider,
            _runtimeManager,
            _executeManager,
            username: userId,
            connectionConfig: GetConnectionConfig(provider)
        );

        await pipeline.InitializeAsync(new TransformerPipelineConfig
        {
            TaskType = TransformerTask.TextGeneration,
            Device = TransformerDevice.Auto
        });

        return pipeline;
    }

    private TransformerConnectionConfig GetConnectionConfig(TransformerModelSource provider)
    {
        return provider switch
        {
            TransformerModelSource.OpenAI => new OpenAIConnectionConfig 
            { 
                ApiKey = GetSecureApiKey("OpenAI"),
                RateLimit = new RateLimitConfig { RequestsPerMinute = 100 }
            },
            TransformerModelSource.Azure => new AzureOpenAIConnectionConfig 
            { 
                Endpoint = GetConfigValue("Azure:Endpoint"),
                ApiKey = GetSecureApiKey("Azure")
            },
            _ => throw new NotSupportedException($"Provider {provider} not configured")
        };
    }
}

Security and Compliance

// Configure security policies
var securityPolicy = new SecurityPolicy
{
    DataClassification = DataClassificationLevel.Confidential,
    RequireModelSigning = true,
    RequireSourceVerification = true,
    Encryption = new EncryptionRequirements
    {
        EncryptAtRest = true,
        EncryptInTransit = true
    }
};

// Apply to pipeline
var securityService = serviceProvider.GetService<ITransformerSecurity>();
await securityService.ApplySecurityPolicyAsync("pipeline-id", securityPolicy);
