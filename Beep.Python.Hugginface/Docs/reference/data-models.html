TransformerPipelineConfig

Pipeline Configuration

/// 
/// Configuration for transformer pipelines
/// 
public class TransformerPipelineConfig : Dictionary
{
    /// 
    /// Gets a configuration value with type conversion
    /// 
    public T GetValue(string key, T defaultValue = default!)
    {
        if (!TryGetValue(key, out var value))
            return defaultValue;
            
        if (value is T directValue)
            return directValue;
            
        try
        {
            return (T)Convert.ChangeType(value, typeof(T));
        }
        catch
        {
            return defaultValue;
        }
    }
    
    /// 
    /// Sets a configuration value
    /// 
    public void SetValue(string key, T value)
    {
        this[key] = value!;
    }
    
    /// 
    /// Validates the configuration
    /// 
    public List Validate()
    {
        var errors = new List();
        
        // Common validation rules
        if (!ContainsKey("api_key") || string.IsNullOrEmpty(GetValue("api_key")))
            errors.Add("API key is required");
            
        var timeout = GetValue("timeout", 60);
        if (timeout <= 0 || timeout > 3600)
            errors.Add("Timeout must be between 1 and 3600 seconds");
            
        return errors;
    }
}

ClassificationResult

Text Classification Result

/// 
/// Result of text classification operation
/// 
public class ClassificationResult
{
    /// 
    /// List of classification predictions
    /// 
    public List Predictions { get; set; } = new();
    
    /// 
    /// The top prediction
    /// 
    public ClassificationPrediction TopPrediction => 
        Predictions.OrderByDescending(p => p.Confidence).FirstOrDefault() ?? 
        new ClassificationPrediction();
    
    /// 
    /// Classification confidence threshold used
    /// 
    public double ConfidenceThreshold { get; set; } = 0.5;
    
    /// 
    /// Whether the classification meets the confidence threshold
    /// 
    public bool IsConfident => TopPrediction.Confidence >= ConfidenceThreshold;
    
    /// 
    /// Model used for classification
    /// 
    public string? ModelName { get; set; }
    
    /// 
    /// Processing time in milliseconds
    /// 
    public long ProcessingTimeMs { get; set; }
}

/// 
/// Request for multimodal AI processing
/// 
public class MultimodalRequest
{
    /// 
    /// Text prompt or instruction
    /// 
    public string TextPrompt { get; set; } = string.Empty;
    
    /// 
    /// List of image files or URLs
    /// 
    public List Images { get; set; } = new();
    
    /// 
    /// List of audio file paths
    /// 
    public List AudioFiles { get; set; } = new();
    
    /// 
    /// List of video file paths
    /// 
    public List VideoFiles { get; set; } = new();
    
    /// 
    /// List of document file paths
    /// 
    public List Documents { get; set; } = new();
    
    /// 
    /// Base64 encoded images
    /// 
    public List ImageData { get; set; } = new();
    
    /// 
    /// Request metadata
    /// 
    public Dictionary Metadata { get; set; } = new();
    
    /// 
    /// Validates the multimodal request
    /// 
    public bool IsValid()
    {
        return !string.IsNullOrEmpty(TextPrompt) || 
               Images.Any() || 
               AudioFiles.Any() || 
               VideoFiles.Any() || 
               Documents.Any() ||
               ImageData.Any();
    }
}

PerformanceMetrics

Performance Metrics

/// 
/// Performance metrics for transformer operations
/// 
public class PerformanceMetrics
{
    /// 
    /// Metric timestamp
    /// 
    public DateTimeOffset Timestamp { get; set; } = DateTimeOffset.UtcNow;
    
    /// 
    /// Request count
    /// 
    public long RequestCount { get; set; }
    
    /// 
    /// Success count
    /// 
    public long SuccessCount { get; set; }
    
    /// 
    /// Error count
    /// 
    public long ErrorCount { get; set; }
    
    /// 
    /// Success rate (percentage)
    /// 
    public double SuccessRate => RequestCount > 0 ? (double)SuccessCount / RequestCount * 100 : 0;
    
    /// 
    /// Average response time in milliseconds
    /// 
    public double AverageResponseTime { get; set; }
    
    /// 
    /// 95th percentile response time
    /// 
    public double P95ResponseTime { get; set; }
    
    /// 
    /// Total tokens processed
    /// 
    public long TotalTokens { get; set; }
    
    /// 
    /// Total cost incurred
    /// 
    public decimal TotalCost { get; set; }
    
    /// 
    /// Average cost per request
    /// 
    public decimal AverageCostPerRequest => RequestCount > 0 ? TotalCost / RequestCount : 0;
    
    /// 
    /// Throughput (requests per second)
    /// 
    public double Throughput { get; set; }
    
    /// 
    /// Error rate (percentage)
    /// 
    public double ErrorRate => RequestCount > 0 ? (double)ErrorCount / RequestCount * 100 : 0;
}
