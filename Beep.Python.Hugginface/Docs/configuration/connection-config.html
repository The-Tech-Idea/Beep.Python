Configuration Overview

The Beep.Python.AI.Transformers library supports multiple AI providers, each with specific connection requirements and authentication methods. This guide covers the complete setup process for each provider.

?? Security Best Practices

  • Never hardcode API keys in your source code
  • Use environment variables or secure configuration stores
  • Rotate keys regularly according to provider recommendations
  • Monitor usage to detect unauthorized access
  • Use least privilege principles for API key permissions

3. Programmatic Configuration

C# Configuration

var openAIConfig = new OpenAIConnectionConfig
{
    ApiKey = Environment.GetEnvironmentVariable("OPENAI_API_KEY"),
    OrganizationId = Environment.GetEnvironmentVariable("OPENAI_ORG_ID"),
    BaseUrl = "https://api.openai.com/v1",
    TimeoutSeconds = 60,
    MaxRetries = 3,
    RateLimit = new RateLimitConfig
    {
        RequestsPerMinute = 500,
        TokensPerMinute = 10000
    }
};

// Register with dependency injection
services.Configure(config =>
{
    config.ApiKey = openAIConfig.ApiKey;
    config.OrganizationId = openAIConfig.OrganizationId;
    config.BaseUrl = openAIConfig.BaseUrl;
    config.TimeoutSeconds = openAIConfig.TimeoutSeconds;
    config.MaxRetries = openAIConfig.MaxRetries;
});

Advanced Connection Settings

1. Connection Pooling

HTTP Client Configuration

// Configure HTTP client with connection pooling
services.AddHttpClient("TransformerPipeline", client =>
{
    client.Timeout = TimeSpan.FromSeconds(120);
    client.DefaultRequestHeaders.Add("User-Agent", "Beep.Python.AI.Transformers/1.0");
})
.ConfigurePrimaryHttpMessageHandler(() => new HttpClientHandler()
{
    MaxConnectionsPerServer = 10,
    PooledConnectionLifetime = TimeSpan.FromMinutes(15),
    PooledConnectionIdleTimeout = TimeSpan.FromMinutes(5)
});

// Global connection settings
services.Configure(options =>
{
    options.DefaultTimeout = TimeSpan.FromSeconds(60);
    options.MaxRetries = 3;
    options.RetryDelay = TimeSpan.FromSeconds(2);
    options.EnableCompression = true;
    options.MaxResponseContentBufferSize = 10 * 1024 * 1024; // 10MB
});

Connection Troubleshooting

1. Common Connection Issues

? Authentication Failures

  • Verify API key format and validity
  • Check key permissions and quotas
  • Ensure organization ID is correct
  • Validate environment variable names

?? Timeout Issues

  • Increase timeout values for large requests
  • Check network connectivity
  • Verify proxy settings
  • Monitor server response times

?? Rate Limiting

  • Implement exponential backoff
  • Monitor rate limit headers
  • Adjust concurrent request limits
  • Consider upgrading API plan

?? Network Issues

  • Test connectivity with curl/Postman
  • Check firewall rules
  • Verify DNS resolution
  • Test without proxy

2. Diagnostic Tools

Connection Testing Utility

public class ConnectionDiagnostics
{
    public async Task TestConnectionAsync(TransformerModelSource provider)
    {
        var result = new DiagnosticResult();
        
        try
        {
            // Test basic connectivity
            result.ConnectivityTest = await TestBasicConnectivity(provider);
            
            // Test authentication
            result.AuthenticationTest = await TestAuthentication(provider);
            
            // Test API availability
            result.ApiAvailabilityTest = await TestApiAvailability(provider);
            
            // Test model access
            result.ModelAccessTest = await TestModelAccess(provider);
            
            // Performance test
            result.PerformanceTest = await TestPerformance(provider);
            
            result.OverallStatus = result.AllTestsPassed ? "PASS" : "FAIL";
        }
        catch (Exception ex)
        {
            result.OverallStatus = "ERROR";
            result.ErrorMessage = ex.Message;
        }
        
        return result;
    }
    
    private async Task TestBasicConnectivity(TransformerModelSource provider)
    {
        using var client = new HttpClient();
        client.Timeout = TimeSpan.FromSeconds(10);
        
        var endpoint = GetProviderEndpoint(provider);
        
        try
        {
            var response = await client.GetAsync($"{endpoint}/health");
            return response.IsSuccessStatusCode;
        }
        catch
        {
            return false;
        }
    }
}

3. Logging and Monitoring

Connection Monitoring

// Enhanced logging configuration
services.AddLogging(builder =>
{
    builder.AddConsole();
    builder.AddFile("logs/connections-{Date}.txt");
    builder.SetMinimumLevel(LogLevel.Information);
    
    builder.AddFilter("Beep.Python.AI.Transformers.Connection", LogLevel.Debug);
});

// Connection metrics
services.AddSingleton();

public class ConnectionMetrics
{
    private readonly ILogger _logger;
    private readonly IMetricsCollector _metrics;
    
    public void RecordConnectionAttempt(string provider, bool success, TimeSpan duration)
    {
        _metrics.Increment($"connection.attempts.{provider}");
        
        if (success)
        {
            _metrics.Increment($"connection.success.{provider}");
            _metrics.RecordValue($"connection.duration.{provider}", duration.TotalMilliseconds);
        }
        else
        {
            _metrics.Increment($"connection.failure.{provider}");
            _logger.LogWarning("Connection failed for provider {Provider} after {Duration}ms", 
                provider, duration.TotalMilliseconds);
        }
    }
}
