MultimodalPipelineFactory Class

Factory class for creating and configuring multimodal transformer pipelines with pre-built templates and enterprise features

?? Overview

The MultimodalPipelineFactory provides a convenient way to create pre-configured multimodal transformer pipelines for common use cases. It offers templates for creative content generation, content analysis, accessibility features, enterprise deployments, and more, eliminating the need for manual configuration in most scenarios.

1. Complete Content Creation Workflow

using Beep.Python.AI.Transformers;

public class ContentCreationService
{
    private readonly IPythonRunTimeManager _runtimeManager;
    private readonly IPythonCodeExecuteManager _executeManager;
    
    public ContentCreationService(
        IPythonRunTimeManager runtimeManager,
        IPythonCodeExecuteManager executeManager)
    {
        _runtimeManager = runtimeManager;
        _executeManager = executeManager;
    }
    
    public async Task<MultimediaContent> CreateSocialMediaContentAsync(
        string topic, 
        string platform = "instagram")
    {
        // Create appropriate pipeline for creative content
        var pipeline = await MultimodalPipelineFactory
            .CreateCreativeContentPipelineAsync(_runtimeManager, _executeManager);
        
        try
        {
            // Generate text content
            var textPrompt = $"Create engaging {platform} post about {topic}";
            var textResult = await pipeline.GenerateTextAsync(textPrompt);
            
            // Generate matching image
            var imagePrompt = $"Create {platform} style image for: {textResult.Data}";
            var imageResult = await pipeline.GenerateImageFromTextAsync(
                imagePrompt,
                new TextToImageParameters 
                { 
                    Width = platform == "instagram" ? 1080 : 1200,
                    Height = platform == "instagram" ? 1080 : 630,
                    Style = "modern",
                    Quality = "high"
                });
            
            // Generate audio if needed (for stories/reels)
            AudioResult? audioResult = null;
            if (platform is "instagram" or "tiktok")
            {
                audioResult = await pipeline.GenerateAudioFromTextAsync(
                    textResult.Data,
                    new TextToAudioParameters 
                    { 
                        Voice = "engaging",
                        Speed = 1.1f,
                        Quality = "high"
                    });
            }
            
            return new MultimediaContent
            {
                Text = textResult.Data,
                Image = imageResult.Data,
                Audio = audioResult,
                Platform = platform,
                Topic = topic
            };
        }
        finally
        {
            pipeline?.Dispose();
        }
    }
}

// Usage
var contentService = new ContentCreationService(runtimeManager, executeManager);
var content = await contentService.CreateSocialMediaContentAsync(
    "sustainable living tips", "instagram");

Console.WriteLine($"Generated content for {content.Platform}:");
Console.WriteLine($"Text: {content.Text}");
Console.WriteLine($"Image: {content.Image.Width}x{content.Image.Height}");
Console.WriteLine($"Audio: {content.Audio?.Duration}s");

2. Enterprise Multi-User Service

public class EnterpriseMultimodalService
{
    private readonly IPythonRunTimeManager _runtimeManager;
    private readonly IPythonCodeExecuteManager _executeManager;
    private readonly ConcurrentDictionary<string, MultimodalTransformerPipeline> _userPipelines;
    
    public EnterpriseMultimodalService(
        IPythonRunTimeManager runtimeManager,
        IPythonCodeExecuteManager executeManager)
    {
        _runtimeManager = runtimeManager;
        _executeManager = executeManager;
        _userPipelines = new ConcurrentDictionary<string, MultimodalTransformerPipeline>();
    }
    
    public async Task<MultimodalTransformerPipeline> GetUserPipelineAsync(
        string username, 
        string? environmentId = null)
    {
        return _userPipelines.GetOrAdd(username, async (user) =>
        {
            // Create session-aware pipeline for user
            var pipeline = await MultimodalPipelineFactory
                .CreateSessionAwarePipelineAsync(
                    _runtimeManager, 
                    _executeManager, 
                    user, 
                    environmentId);
            
            // Configure for enterprise use
            return pipeline;
        });
    }
    
    public async Task<string> ProcessDocumentAsync(
        string username, 
        string documentPath, 
        string analysisType = "summary")
    {
        var pipeline = await GetUserPipelineAsync(username);
        
        switch (analysisType.ToLower())
        {
            case "summary":
                var description = await pipeline.GenerateTextFromImageAsync(
                    documentPath,
                    new ImageToTextParameters 
                    { 
                        Style = "professional",
                        DetailedAnalysis = true
                    });
                return description.Data;
                
            case "qa":
                var answer = await pipeline.AnswerQuestionAboutImageAsync(
                    documentPath,
                    "What are the key points in this document?");
                return answer.Data;
                
            default:
                throw new ArgumentException($"Unknown analysis type: {analysisType}");
        }
    }
    
    public async Task CleanupUserSessionAsync(string username)
    {
        if (_userPipelines.TryRemove(username, out var pipeline))
        {
            pipeline.Dispose();
        }
    }
    
    public async Task CleanupAllSessionsAsync()
    {
        var pipelines = _userPipelines.Values.ToList();
        _userPipelines.Clear();
        
        foreach (var pipeline in pipelines)
        {
            pipeline.Dispose();
        }
    }
}

// Usage in enterprise application
var enterpriseService = new EnterpriseMultimodalService(runtimeManager, executeManager);

// Process documents for multiple users concurrently
var users = new[] { "alice@company.com", "bob@company.com", "carol@company.com" };
var tasks = users.Select(async user =>
{
    var summary = await enterpriseService.ProcessDocumentAsync(
        user, 
        $"documents/{user}_report.pdf", 
        "summary");
    
    return new { User = user, Summary = summary };
});

var results = await Task.WhenAll(tasks);
foreach (var result in results)
{
    Console.WriteLine($"{result.User}: {result.Summary}");
}

// Cleanup when done
await enterpriseService.CleanupAllSessionsAsync();

? Best Practices

??? Factory Method Selection

  • Quick Start: Use pre-built templates for common scenarios
  • Specialized: Choose domain-specific factories for targeted use cases
  • Enterprise: Always use session-aware methods in multi-user environments
  • Lightweight: Use minimal pipelines for single-task applications

?? Resource Management

  • Disposal: Always dispose pipelines when done to free resources
  • Session Cleanup: Clean up user sessions in enterprise environments
  • Memory Monitoring: Monitor memory usage with high-quality models
  • Concurrent Limits: Respect MaxConcurrentTasks to avoid overload

? Performance Optimization

  • Pipeline Reuse: Reuse pipelines across multiple operations
  • Model Caching: Enable caching for frequently used models
  • Quality Tuning: Choose appropriate quality levels for your use case
  • Batch Processing: Process multiple items together when possible
