Quick Start Example

Here's the simplest way to get started with text generation using OpenAI:

Complete Example

using Beep.Python.AI.Transformers;
using Beep.Python.Model;

public class QuickStartExample
{
    private readonly IPythonRunTimeManager _runtimeManager;
    private readonly IPythonCodeExecuteManager _executeManager;

    public QuickStartExample(
        IPythonRunTimeManager runtimeManager,
        IPythonCodeExecuteManager executeManager)
    {
        _runtimeManager = runtimeManager;
        _executeManager = executeManager;
    }

    public async Task<string> GenerateTextAsync(string prompt)
    {
        // 1. Configure OpenAI connection
        var openAIConfig = new OpenAIConnectionConfig
        {
            ApiKey = "sk-your-openai-api-key-here",
            TimeoutSeconds = 60,
            MaxRetries = 3
        };

        // 2. Create pipeline
        var pipeline = TransformerPipelineFactory.CreateOpenAIPipeline(
            _runtimeManager,
            _executeManager,
            openAIConfig
        );

        try
        {
            // 3. Initialize pipeline
            await pipeline.InitializeAsync(new TransformerPipelineConfig
            {
                TaskType = TransformerTask.TextGeneration,
                Device = TransformerDevice.Auto
            });

            // 4. Load model
            await pipeline.LoadModelAsync(new TransformerModelInfo
            {
                Name = "gpt-4",
                Source = TransformerModelSource.OpenAI
            }, TransformerTask.TextGeneration);

            // 5. Generate text
            var result = await pipeline.GenerateTextAsync(prompt);
            
            return result.Success ? result.Data : $"Error: {result.ErrorMessage}";
        }
        finally
        {
            // 6. Clean up
            pipeline.Dispose();
        }
    }
}

// Usage
var example = new QuickStartExample(runtimeManager, executeManager);
var response = await example.GenerateTextAsync("Write a haiku about programming");
Console.WriteLine(response);

Text Classification

public class TextClassificationExample
{
    public async Task RunClassificationAsync()
    {
        // Using HuggingFace for classification
        var pipeline = TransformerPipelineFactory.CreatePipeline(
            TransformerModelSource.HuggingFace,
            runtimeManager,
            executeManager
        );

        await pipeline.InitializeAsync(new TransformerPipelineConfig
        {
            TaskType = TransformerTask.TextClassification
        });

        await pipeline.LoadModelAsync(new TransformerModelInfo
        {
            Name = "cardiffnlp/twitter-roberta-base-sentiment-latest",
            Source = TransformerModelSource.HuggingFace
        }, TransformerTask.TextClassification);

        var texts = new[]
        {
            "I love this product! It's amazing!",
            "This service is terrible and slow.",
            "The weather is okay today."
        };

        foreach (var text in texts)
        {
            var result = await pipeline.ClassifyTextAsync(text);
            if (result.Success)
            {
                Console.WriteLine($"Text: {text}");
                foreach (var label in result.Data.Labels)
                {
                    Console.WriteLine($"  {label.Label}: {label.Score:F3}");
                }
                Console.WriteLine();
            }
        }
    }
}

Text Embeddings

public class EmbeddingsExample
{
    public async Task RunEmbeddingsAsync()
    {
        // Using OpenAI for embeddings
        var config = new OpenAIConnectionConfig
        {
            ApiKey = "sk-your-openai-api-key"
        };

        var pipeline = TransformerPipelineFactory.CreateOpenAIPipeline(
            runtimeManager, executeManager, config);

        await pipeline.InitializeAsync(new TransformerPipelineConfig
        {
            TaskType = TransformerTask.FeatureExtraction
        });

        await pipeline.LoadModelAsync(new TransformerModelInfo
        {
            Name = "text-embedding-ada-002",
            Source = TransformerModelSource.OpenAI
        }, TransformerTask.FeatureExtraction);

        var texts = new List<string>
        {
            "The quick brown fox jumps over the lazy dog.",
            "Artificial intelligence is transforming the world.",
            "Python is a versatile programming language."
        };

        var result = await pipeline.GetEmbeddingsAsync(texts);
        
        if (result.Success)
        {
            Console.WriteLine("Generated embeddings:");
            for (int i = 0; i < texts.Count; i++)
            {
                Console.WriteLine($"Text: {texts[i]}");
                Console.WriteLine($"Embedding dimensions: {result.Data[i].Length}");
                Console.WriteLine($"First 5 values: [{string.Join(", ", result.Data[i].Take(5).Select(x => x.ToString("F4")))}]");
                Console.WriteLine();
            }

            // Calculate similarity between first two embeddings
            var similarity = CosineSimilarity(result.Data[0], result.Data[1]);
            Console.WriteLine($"Similarity between first two texts: {similarity:F4}");
        }
    }

    private double CosineSimilarity(float[] vectorA, float[] vectorB)
    {
        var dotProduct = vectorA.Zip(vectorB, (a, b) => a * b).Sum();
        var magnitudeA = Math.Sqrt(vectorA.Sum(a => a * a));
        var magnitudeB = Math.Sqrt(vectorB.Sum(b => b * b));
        return dotProduct / (magnitudeA * magnitudeB);
    }
}

Error Handling and Best Practices

Comprehensive Error Handling

public class RobustAIService
{
    private readonly ILogger<RobustAIService> _logger;
    private readonly IPythonRunTimeManager _runtimeManager;
    private readonly IPythonCodeExecuteManager _executeManager;

    public RobustAIService(
        ILogger<RobustAIService> logger,
        IPythonRunTimeManager runtimeManager,
        IPythonCodeExecuteManager executeManager)
    {
        _logger = logger;
        _runtimeManager = runtimeManager;
        _executeManager = executeManager;
    }

    public async Task<string> SafeGenerateTextAsync(string prompt, int maxRetries = 3)
    {
        ITransformerPipeLine? pipeline = null;
        
        try
        {
            pipeline = await CreatePipelineWithRetryAsync(maxRetries);
            
            // Subscribe to events for monitoring
            pipeline.ErrorOccurred += OnErrorOccurred;
            pipeline.InferenceStarted += OnInferenceStarted;
            pipeline.InferenceCompleted += OnInferenceCompleted;

            var result = await pipeline.GenerateTextAsync(prompt);
            
            if (result.Success)
            {
                _logger.LogInformation("Text generation successful. Execution time: {ExecutionTime}ms", result.ExecutionTimeMs);
                return result.Data;
            }
            else
            {
                _logger.LogWarning("Text generation failed: {Error}", result.ErrorMessage);
                return "Sorry, I couldn't generate a response at this time.";
            }
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Unexpected error during text generation");
            return "An unexpected error occurred. Please try again later.";
        }
        finally
        {
            pipeline?.Dispose();
        }
    }

    private async Task<ITransformerPipeLine> CreatePipelineWithRetryAsync(int maxRetries)
    {
        Exception? lastException = null;
        
        for (int attempt = 1; attempt <= maxRetries; attempt++)
        {
            try
            {
                var config = new OpenAIConnectionConfig
                {
                    ApiKey = GetApiKey(),
                    TimeoutSeconds = 60 * attempt, // Increase timeout with retries
                    MaxRetries = 1
                };

                var pipeline = TransformerPipelineFactory.CreateOpenAIPipeline(
                    _runtimeManager, _executeManager, config);

                await pipeline.InitializeAsync(new TransformerPipelineConfig
                {
                    TaskType = TransformerTask.TextGeneration
                });

                await pipeline.LoadModelAsync(new TransformerModelInfo
                {
                    Name = "gpt-4",
                    Source = TransformerModelSource.OpenAI
                }, TransformerTask.TextGeneration);

                _logger.LogInformation("Pipeline created successfully on attempt {Attempt}", attempt);
                return pipeline;
            }
            catch (Exception ex)
            {
                lastException = ex;
                _logger.LogWarning("Pipeline creation failed on attempt {Attempt}: {Error}", attempt, ex.Message);
                
                if (attempt < maxRetries)
                {
                    var delay = TimeSpan.FromSeconds(Math.Pow(2, attempt)); // Exponential backoff
                    _logger.LogInformation("Retrying in {Delay} seconds...", delay.TotalSeconds);
                    await Task.Delay(delay);
                }
            }
        }

        throw new InvalidOperationException($"Failed to create pipeline after {maxRetries} attempts", lastException);
    }

    private void OnErrorOccurred(object? sender, TransformerErrorEventArgs args)
    {
        _logger.LogError(args.Exception, "Pipeline error: {Message}", args.ErrorMessage);
    }

    private void OnInferenceStarted(object? sender, TransformerEventArgs args)
    {
        _logger.LogDebug("Inference started for model: {ModelName}", args.ModelName);
    }

    private void OnInferenceCompleted(object? sender, TransformerEventArgs args)
    {
        _logger.LogDebug("Inference completed for model: {ModelName}", args.ModelName);
    }

    private string GetApiKey()
    {
        return Environment.GetEnvironmentVariable("OPENAI_API_KEY") 
               ?? throw new InvalidOperationException("OpenAI API key not found");
    }
}

Connection Health Monitoring

public class HealthMonitoringService
{
    private readonly Dictionary<TransformerModelSource, ITransformerPipeLine> _pipelines = new();
    private readonly Timer _healthCheckTimer;

    public HealthMonitoringService()
    {
        _healthCheckTimer = new Timer(PerformHealthCheck, null, 
            TimeSpan.FromMinutes(1), TimeSpan.FromMinutes(5));
    }

    public async Task<bool> IsProviderHealthyAsync(TransformerModelSource provider)
    {
        if (!_pipelines.ContainsKey(provider))
            return false;

        try
        {
            var pipeline = _pipelines[provider];
            
            // Simple health check with a basic prompt
            var result = await pipeline.GenerateTextAsync("Hello", new TextGenerationParameters
            {
                MaxTokens = 1,
                Temperature = 0
            });

            return result.Success;
        }
        catch (Exception ex)
        {
            Console.WriteLine($"Health check failed for {provider}: {ex.Message}");
            return false;
        }
    }

    private async void PerformHealthCheck(object? state)
    {
        foreach (var provider in _pipelines.Keys)
        {
            var isHealthy = await IsProviderHealthyAsync(provider);
            Console.WriteLine($"Provider {provider}: {(isHealthy ? "Healthy" : "Unhealthy")}");
        }
    }
}

Performance Tips and Best Practices

1. Pipeline Reuse

✅ Do: Reuse pipeline instances for multiple requests
❌ Don't: Create new pipeline instances for each request 
// ✅ Good: Singleton pipeline service
public class PipelineService
{
    private readonly ITransformerPipeLine _pipeline;

    public PipelineService(ITransformerPipeLine pipeline)
    {
        _pipeline = pipeline;
    }

    public async Task<string> GenerateAsync(string prompt)
    {
        var result = await _pipeline.GenerateTextAsync(prompt);
        return result.Data;
    }
}

// ❌ Bad: Creating new pipeline each time
public async Task<string> GenerateAsync(string prompt)
{
    using var pipeline = TransformerPipelineFactory.CreateOpenAIPipeline(/*...*/);
    // ... initialization code
    var result = await pipeline.GenerateTextAsync(prompt);
    return result.Data;
}
