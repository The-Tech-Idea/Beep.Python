?? Setup and Configuration

1. Basic Initialization

using Beep.Python.DataManagement;

// Initialize the pandas manager
var pandasManager = new PythonPandasManager(beepService, runtimeManager, executeManager);

// Configure for your user (simplest approach)
bool configured = pandasManager.ConfigureSessionForUser("john.doe");

if (configured)
{
    Console.WriteLine("? Ready to work with pandas DataFrames!");
}
else
{
    Console.WriteLine("? Failed to configure session");
}

2. Dependency Injection Setup

// In Program.cs or Startup.cs
builder.Services.AddTransient<IPythonPandasManager, PythonPandasManager>();

// In your controller or service
public class DataController : ControllerBase
{
    private readonly IPythonPandasManager _pandasManager;
    
    public DataController(IPythonPandasManager pandasManager)
    {
        _pandasManager = pandasManager;
    }
    
    [HttpGet("analyze")]
    public IActionResult AnalyzeData()
    {
        // Configure session for current user
        _pandasManager.ConfigureSessionForUser(User.Identity.Name);
        
        // Your data operations here...
        return Ok("Analysis complete");
    }
}

3. Using Statement for Cleanup

// Automatic resource cleanup
using var pandasManager = new PythonPandasManager(beepService, runtimeManager, executeManager);

// Configure and use
pandasManager.ConfigureSessionForUser("analyst");

// Load some data
pandasManager.ReadCsv("sales", @"C:\Data\sales.csv");

// Get summary
string summary = pandasManager.Describe("sales");
Console.WriteLine($"Data Summary:\n{summary}");

// pandasManager.Dispose() called automatically when exiting using block

?? Loading and Saving Data

1. Load CSV Files

// Load a CSV file
pandasManager.ReadCsv("sales_data", @"C:\Data\sales_2024.csv");

// Check what we loaded
string info = pandasManager.Describe("sales_data");
Console.WriteLine($"Loaded data summary:\n{info}");

// Async loading for large files
await pandasManager.ReadCsvAsync("large_dataset", @"C:\BigData\huge_file.csv");

2. Load Excel Files

// Load Excel file
pandasManager.ReadExcel("product_catalog", @"C:\Data\products.xlsx");

// Get unique product categories
string categories = pandasManager.UniqueValues("product_catalog", "Category");
Console.WriteLine($"Product categories: {categories}");

3. Save Data in Different Formats

// Save as CSV
pandasManager.ToCsv("sales_data", @"C:\Output\processed_sales.csv");

// Save as Excel
pandasManager.ToExcel("sales_data", @"C:\Output\sales_report.xlsx");

// Save as JSON
pandasManager.ToJson("sales_data", @"C:\Output\sales_data.json");

// Generic save method
pandasManager.ExportDataFrame("sales_data", @"C:\Output\data.parquet", "parquet");

4. Load from Database

// Load data from SQL Server
string connectionString = "Server=localhost;Database=SalesDB;Trusted_Connection=true;";
string query = "SELECT * FROM Orders WHERE OrderDate >= '2024-01-01'";

pandasManager.ReadSql("recent_orders", query, connectionString);

// Check how many orders we got
string orderCount = pandasManager.CalculateMean("recent_orders", "OrderID");
Console.WriteLine($"Loaded {orderCount} recent orders");

?? Filtering and Selecting Data

1. Filter Rows by Condition

// Filter high-value sales (amount > 1000)
pandasManager.FilterRows("sales_data", "high_value_sales", "Amount > 1000");

// Filter by multiple conditions
pandasManager.FilterRows("sales_data", "premium_recent", 
    "(Amount > 500) & (OrderDate >= '2024-01-01')");

// Filter by string values
pandasManager.FilterRows("customers", "vip_customers", "CustomerType == 'VIP'");

2. Select Specific Columns

// Select just the columns we need
string[] importantColumns = { "CustomerName", "Amount", "OrderDate", "Region" };
pandasManager.SelectColumns("sales_data", "sales_summary", importantColumns);

// Work with the summarized data
string summary = pandasManager.Describe("sales_summary");
Console.WriteLine(summary);

3. Sample Data for Testing

// Get a 10% random sample for testing
pandasManager.SampleDataFrame("large_dataset", "test_sample", 0.1);

// Use the sample for quick analysis
string testStats = pandasManager.Describe("test_sample");
Console.WriteLine($"Sample statistics:\n{testStats}");

4. Query with pandas Query Syntax

// Use pandas query syntax for complex filtering
pandasManager.QueryDataFrame("sales_data", "filtered_sales", 
    "Amount > 100 and Region in ['North', 'South']");

// Date-based queries
pandasManager.QueryDataFrame("orders", "recent_orders", 
    "OrderDate >= '2024-01-01' and OrderDate < '2024-02-01'");

?? Basic Statistics and Analysis

1. Descriptive Statistics

// Get comprehensive statistics
string overallStats = pandasManager.Describe("sales_data");
Console.WriteLine("Overall Statistics:");
Console.WriteLine(overallStats);

// Individual statistics
string avgSales = pandasManager.CalculateMean("sales_data", "Amount");
string medianSales = pandasManager.CalculateMedian("sales_data", "Amount");

Console.WriteLine($"Average Sale: ${avgSales}");
Console.WriteLine($"Median Sale: ${medianSales}");

2. Data Exploration

// Check for missing values
string missingData = pandasManager.IsNull("sales_data");
Console.WriteLine("Missing values by column:");
Console.WriteLine(missingData);

// Get unique values in categorical columns
string regions = pandasManager.UniqueValues("sales_data", "Region");
string customerTypes = pandasManager.UniqueValues("sales_data", "CustomerType");

Console.WriteLine($"Regions: {regions}");
Console.WriteLine($"Customer Types: {customerTypes}");

3. Value Counts and Distribution

// See distribution of categorical data
string regionCounts = pandasManager.ValueCounts("sales_data", "Region");
string productCounts = pandasManager.ValueCounts("sales_data", "ProductCategory");

Console.WriteLine("Sales by Region:");
Console.WriteLine(regionCounts);

Console.WriteLine("\nSales by Product Category:");
Console.WriteLine(productCounts);

4. Correlation Analysis

// Check correlations between numeric columns
string correlations = pandasManager.Correlation("sales_data");
Console.WriteLine("Correlation Matrix:");
Console.WriteLine(correlations);

// This helps identify relationships between variables like:
// - Price vs Quantity
// - Customer Age vs Purchase Amount
// - etc.

?? Basic Data Cleaning

1. Handle Missing Values

// Remove rows with any missing values
pandasManager.DropNA("raw_data", "clean_data");

// Fill missing values with defaults
pandasManager.FillNA("customer_data", "complete_data", "Unknown");

// Fill numeric columns with 0
pandasManager.FillNA("sales_data", "filled_sales", 0);

2. Remove Duplicates

// Remove duplicate rows
pandasManager.DropDuplicates("customer_list", "unique_customers");

// Check the difference
Console.WriteLine("Duplicates removed successfully");

3. Add Calculated Columns

// Add a profit column
pandasManager.AddColumn("sales_data", "Profit", "Revenue - Cost");

// Add a profit margin percentage
pandasManager.AddColumn("sales_data", "ProfitMargin", "(Profit / Revenue) * 100");

// Add a customer category based on purchase amount
pandasManager.AddColumn("sales_data", "CustomerSegment", 
    "np.where(Amount > 1000, 'Premium', 'Standard')");

?? Basic Grouping and Aggregation

1. Group by Single Column

// Total sales by region
pandasManager.GroupBy("sales_data", "sales_by_region", "Region", "sum");

// Average order size by customer type
pandasManager.GroupBy("sales_data", "avg_by_customer_type", "CustomerType", "mean");

// Count of orders by month
pandasManager.GroupBy("sales_data", "orders_by_month", "Month", "count");

2. Export Grouped Results

// Save regional analysis
pandasManager.ToCsv("sales_by_region", @"C:\Reports\regional_analysis.csv");

// Convert to JSON for web display
string jsonData = pandasManager.GetDataFrameAsJson("sales_by_region");
Console.WriteLine("Regional data as JSON:");
Console.WriteLine(jsonData);

?? Complete Basic Example

Simple Sales Analysis Workflow

using Beep.Python.DataManagement;

public class SimpleSalesAnalysis
{
    public static void AnalyzeSalesData()
    {
        // 1. Setup
        using var pandasManager = new PythonPandasManager(beepService, runtimeManager, executeManager);
        pandasManager.ConfigureSessionForUser("analyst");

        try
        {
            Console.WriteLine("?? Starting Sales Analysis...");

            // 2. Load data
            pandasManager.ReadCsv("sales", @"C:\Data\sales_2024.csv");
            Console.WriteLine("? Data loaded successfully");

            // 3. Quick data check
            string summary = pandasManager.Describe("sales");
            Console.WriteLine($"?? Data Summary:\n{summary}");

            // 4. Check for data quality issues
            string missing = pandasManager.IsNull("sales");
            Console.WriteLine($"?? Missing values:\n{missing}");

            // 5. Clean the data
            pandasManager.DropNA("sales", "clean_sales");
            pandasManager.DropDuplicates("clean_sales", "final_sales");
            Console.WriteLine("?? Data cleaned");

            // 6. Basic analysis
            string avgSale = pandasManager.CalculateMean("final_sales", "Amount");
            string regions = pandasManager.ValueCounts("final_sales", "Region");
            
            Console.WriteLine($"?? Average Sale: ${avgSale}");
            Console.WriteLine($"??? Sales by Region:\n{regions}");

            // 7. Group by region for summary
            pandasManager.GroupBy("final_sales", "regional_summary", "Region", "sum");

            // 8. Save results
            pandasManager.ToCsv("regional_summary", @"C:\Reports\regional_analysis.csv");
            pandasManager.ToExcel("final_sales", @"C:\Reports\clean_sales_data.xlsx");

            Console.WriteLine("? Analysis complete! Check C:\\Reports\\ for results.");
        }
        catch (Exception ex)
        {
            Console.WriteLine($"? Error during analysis: {ex.Message}");
        }
    }
}

?? Tips for Beginners

?? Start Simple

  • Begin with small CSV files to test your setup
  • Use Describe() to understand your data first
  • Check for missing values with IsNull() before analysis
  • Always configure your session before operations

?? Data Exploration

  • Use UniqueValues() to see categorical data options
  • Use ValueCounts() to understand data distribution
  • Sample large datasets first: SampleDataFrame()
  • Save intermediate results to avoid re-processing

?? Common Mistakes

  • Forgetting to configure the session first
  • Not checking for missing values before analysis
  • Using invalid pandas column names in conditions
  • Not disposing the manager properly
