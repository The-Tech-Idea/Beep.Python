?? Migration Guides

?? Upgrading from 0.x to 1.0+

The 1.0 release introduces significant API changes. Follow this guide for smooth migration:

1. Constructor Changes
// ? OLD (0.x)
var manager = new PythonPandasManager();
manager.Initialize(); // Basic initialization

// ? NEW (1.0+)
var manager = new PythonPandasManager(beepService, runtimeManager, executeManager);
manager.ConfigureSessionForUser("username"); // Required session configuration
2. Dependency Injection Setup
// ? NEW in 1.0 - Recommended approach
services.AddSingleton<IBeepService, BeepService>();
services.AddSingleton<IPythonRunTimeManager, PythonRunTimeManager>();
services.AddSingleton<IPythonCodeExecuteManager, PythonCodeExecuteManager>();
services.AddTransient<IPythonPandasManager, PythonPandasManager>();

// Use in your classes
public class DataService
{
    private readonly IPythonPandasManager _pandasManager;
    
    public DataService(IPythonPandasManager pandasManager)
    {
        _pandasManager = pandasManager;
    }
    
    public void ProcessData()
    {
        _pandasManager.ConfigureSessionForUser("current-user");
        // Your data operations...
    }
}
3. Async Operations
// ? NEW in 1.0 - Use async for large operations
await pandasManager.ReadCsvAsync("data", "large_file.csv");
await pandasManager.ToCsvAsync("results", "output.csv");
await pandasManager.GroupByAsync("data", "grouped", "category", "sum");
4. Session Management
// ? Always configure session before operations
if (!pandasManager.IsSessionConfigured())
{
    bool configured = pandasManager.ConfigureSessionForUser("username");
    if (!configured)
    {
        throw new InvalidOperationException("Failed to configure session");
    }
}

// Now safe to perform operations
pandasManager.ReadCsv("data", "file.csv");
5. Resource Management
// ? Use using statements for proper cleanup
using var pandasManager = new PythonPandasManager(beepService, runtimeManager, executeManager);
pandasManager.ConfigureSessionForUser("username");

// Perform operations
pandasManager.ReadCsv("data", "file.csv");
string results = pandasManager.Describe("data");

// Automatic cleanup when exiting using block
