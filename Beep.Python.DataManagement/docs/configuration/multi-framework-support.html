?? Framework-Specific Dependency Injection

1. .NET 6 Configuration

// Program.cs for .NET 6
using Beep.Python.DataManagement;
using Microsoft.Extensions.DependencyInjection;
using Microsoft.Extensions.Hosting;

var builder = WebApplication.CreateBuilder(args);

// .NET 6 specific service registration
builder.Services.AddControllers();
builder.Services.AddLogging();

// Configure Python DataManagement with .NET 6 optimizations
builder.Services.AddPythonDataManagement(options =>
{
    options.PythonDllPath = builder.Configuration["Python:DllPath"];
    options.PythonHomePath = builder.Configuration["Python:HomePath"];
    options.EnableNet6Optimizations = true;
    options.UseMinimalApis = true;
});

// .NET 6 specific middleware
builder.Services.AddScoped<IPythonPandasManager, PythonPandasManager>();

var app = builder.Build();

// .NET 6 middleware pipeline
if (app.Environment.IsDevelopment())
{
    app.UseDeveloperExceptionPage();
}

app.UseHttpsRedirection();
app.UseAuthorization();
app.MapControllers();

// .NET 6 minimal API endpoints
app.MapGet("/api/pandas/status", async (IPythonPandasManager pandasManager) =>
{
    var isConfigured = pandasManager.IsSessionConfigured();
    return Results.Ok(new { configured = isConfigured, framework = "NET6" });
});

await app.RunAsync();

2. .NET 8 Configuration with Native AOT

// Program.cs for .NET 8 with Native AOT support
using Beep.Python.DataManagement;
using Microsoft.Extensions.DependencyInjection;
using System.Text.Json.Serialization;

var builder = WebApplication.CreateSlimBuilder(args);

// .NET 8 specific optimizations
builder.Services.ConfigureHttpJsonOptions(options =>
{
    options.SerializerOptions.TypeInfoResolver = JsonTypeInfoResolver.Combine(
        PandasJsonContext.Default,
        JsonSerializerOptions.Default.TypeInfoResolver);
});

// Enhanced service registration for .NET 8
builder.Services.AddPythonDataManagement(options =>
{
    options.PythonDllPath = builder.Configuration["Python:DllPath"];
    options.PythonHomePath = builder.Configuration["Python:HomePath"];
    
    // .NET 8 specific features
    options.EnableNativeAOT = true;
    options.UseSourceGeneratedJson = true;
    options.EnableMemoryOptimizations = true;
    options.UseEnhancedGarbageCollection = true;
});

// .NET 8 enhanced service lifetimes
builder.Services.AddKeyedSingleton<IPythonPandasManager>("default", 
    (provider, key) => new PythonPandasManager(
        provider.GetRequiredService<IPythonRunTimeManager>(),
        provider.GetRequiredService<IPythonCodeExecuteManager>()));

var app = builder.Build();

// .NET 8 optimized minimal APIs with source generation
app.MapGet("/api/pandas/dataframes", async (IPythonPandasManager pandasManager) =>
{
    var dataframes = await pandasManager.GetActiveDataFramesAsync();
    return TypedResults.Ok(dataframes);
});

app.MapPost("/api/pandas/create", async (CreateDataFrameRequest request, 
    [FromKeyedServices("default")] IPythonPandasManager pandasManager) =>
{
    await pandasManager.CreateDataFrameAsync(request.Name, request.Data);
    return TypedResults.Created($"/api/pandas/dataframes/{request.Name}", request);
});

await app.RunAsync();

// Program.cs for .NET 9 with cutting-edge features
using Beep.Python.DataManagement;

builder.Services.AddPythonDataManagement(options =>

    options.PythonHomePath = builder.Configuration["Python:HomePath"];
    
    // .NET 9 specific enhancements
    options.EnableNet9Optimizations = true;
    options.UseEnhancedPipelineProcessing = true;
    options.EnableAdvancedMemoryManagement = true;
    options.UseImprovedAsyncPatterns = true;
});

// .NET 9 AI integration for enhanced analytics
builder.Services.AddAIServices(ai =>
{
    ai.AddPandasIntegration();
    ai.EnableSmartDataProfiling();
});

// Enhanced performance monitoring
builder.Services.AddTelemetry(telemetry =>
{
    telemetry.AddPandasMetrics();
    telemetry.EnableFrameworkSpecificOptimizations();
});

var app = builder.Build();

// .NET 9 enhanced minimal APIs with improved performance
app.MapGroup("/api/pandas")
    .WithTags("Pandas Operations")
    .WithOpenApi()
    .MapPandasEndpoints(); // Extension method for cleaner organization

// Advanced pipeline endpoint with .NET 9 features
app.MapPost("/api/analytics/pipeline", async (
    AnalyticsPipelineRequest request,
    IPythonPandasManager pandasManager,
    ILogger<Program> logger,
    CancellationToken cancellationToken) =>
{
    using var activity = System.Diagnostics.Activity.StartActivity("AnalyticsPipeline");
    
    try
    {
        // .NET 9 enhanced async processing
        var result = await pandasManager.ExecuteAnalyticsPipelineAsync(
            request, 
            cancellationToken);
        
        return TypedResults.Ok(result);
    }
    catch (OperationCanceledException)
    {
        return TypedResults.Problem("Operation was cancelled", statusCode: 499);
    }
    catch (Exception ex)
    {
        logger.LogError(ex, "Analytics pipeline failed");
        return TypedResults.Problem("Pipeline execution failed");
    }
})
.WithName("ExecuteAnalyticsPipeline")
.WithSummary("Execute advanced analytics pipeline with .NET 9 optimizations");

await app.RunAsync();
