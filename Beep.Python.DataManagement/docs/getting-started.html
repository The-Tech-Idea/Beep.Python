Getting Started

Learn how to set up and use Beep.Python.DataManagement for powerful pandas DataFrame operations in your .NET applications

? Quick Start (5 Minutes)

Get up and running with PythonPandasManager in just a few minutes:

1. Install and Initialize

// Add to your dependency injection container
services.AddSingleton<IPythonRunTimeManager, PythonRunTimeManager>();
services.AddSingleton<IPythonCodeExecuteManager, PythonCodeExecuteManager>();
services.AddTransient<IPythonPandasManager, PythonPandasManager>();

// Or create manually
var pandasManager = new PythonPandasManager(beepService, runtimeManager, executeManager);

2. Configure Session

// Configure for your user
pandasManager.ConfigureSessionForUser("your-username", "data-env");

// Verify configuration
if (pandasManager.IsSessionConfigured()) {
    Console.WriteLine("Ready to analyze data!");
}

3. Load and Analyze Data

// Load your data
pandasManager.ReadCsv("sales_data", @"C:\Data\sales.csv");

// Get quick insights
string summary = pandasManager.Describe("sales_data");
Console.WriteLine(summary);

// Save results
pandasManager.ToCsv("sales_data", @"C:\Output\processed_sales.csv");
?? That's it! You're now ready to perform advanced data analysis with pandas in your .NET application.

?? Basic Setup

Dependency Injection Setup

// In your Startup.cs or Program.cs
public void ConfigureServices(IServiceCollection services)
{
    // Register BEEP services
    services.AddSingleton<IBeepService, BeepService>();
    
    // Register Python runtime services
    services.AddSingleton<IPythonRunTimeManager, PythonRunTimeManager>();
    services.AddSingleton<IPythonCodeExecuteManager, PythonCodeExecuteManager>();
    services.AddSingleton<IPythonSessionManager, PythonSessionManager>();
    services.AddSingleton<IPythonVirtualEnvManager, PythonVirtualEnvManager>();
    
    // Register DataManagement services
    services.AddTransient<IPythonPandasManager, PythonPandasManager>();
}

Manual Initialization

// Create services manually if not using DI
var beepService = new BeepService();
var runtimeManager = new PythonRunTimeManager(beepService);
var executeManager = new PythonCodeExecuteManager(beepService, runtimeManager);

// Initialize runtime with Python configuration
var pythonConfig = new PythonRunTime
{
    PythonDLLPath = @"C:\Python39\python39.dll",
    PythonHomePath = @"C:\Python39",
    VirtualEnvironmentPath = @"C:\Environments\data-env"
};

bool initialized = runtimeManager.Initialize(pythonConfig, null, null, PythonEngineMode.MultiUser);
if (!initialized) {
    throw new InvalidOperationException("Failed to initialize Python runtime");
}

// Create pandas manager
var pandasManager = new PythonPandasManager(beepService, runtimeManager, executeManager);

?? Session Configuration

Why Session Management Matters

PythonPandasManager uses session-based architecture to provide:

  • Multi-user Isolation: Each user gets their own Python environment
  • Resource Management: Automatic cleanup and memory management
  • Security: Data separation between users and sessions
  • Scalability: Support for concurrent data operations

Method 1: Auto-Configure for User (Recommended)

// Let the system handle session creation
bool configured = pandasManager.ConfigureSessionForUser("alice.smith");

if (configured) {
    Console.WriteLine("Session configured successfully!");
    
    // Start working with data
    pandasManager.ReadCsv("my_data", @"C:\Data\sample.csv");
} else {
    Console.WriteLine("Failed to configure session");
}

Method 2: Manual Session Configuration

// Create specific session and environment
var sessionManager = runtimeManager.SessionManager;
var envManager = runtimeManager.VirtualEnvmanager;

// Create or get environment
var environment = envManager.GetEnvironmentById("data-analysis-env") 
                  ?? envManager.CreateEnvironmentForUser(pythonConfig, envBasePath, "alice.smith", "data-env");

// Create session
var session = sessionManager.CreateSession("alice.smith", environment.ID);

// Configure pandas manager
bool configured = pandasManager.ConfigureSession(session, environment);

if (configured) {
    Console.WriteLine($"Session {session.SessionId} configured for user {session.Username}");
}
?? Best Practice: Use ConfigureSessionForUser() for most scenarios. The system will automatically create or reuse appropriate sessions and environments.

?? Your First Data Operations

1. Loading Data from Different Sources

// Ensure session is configured
if (!pandasManager.IsSessionConfigured()) {
    pandasManager.ConfigureSessionForUser("data-analyst");
}

// Load from CSV (most common)
pandasManager.ReadCsv("sales", @"C:\Data\sales_2024.csv");

// Load from Excel
pandasManager.ReadExcel("products", @"C:\Data\product_catalog.xlsx");

// Load from database
string connectionString = "Server=localhost;Database=SalesDB;Trusted_Connection=true;";

    "SELECT CustomerID, Name, Region, TotalSpent FROM Customers", 
    connectionString);

// Load from web API

2. Basic Data Exploration

// Get overview of your data
string salesSummary = pandasManager.Describe("sales");
Console.WriteLine("Sales Data Summary:");
Console.WriteLine(salesSummary);

// Check data types and missing values
string missingValues = pandasManager.IsNull("sales");
Console.WriteLine("\nMissing Values:");
Console.WriteLine(missingValues);

// Get unique values in categorical columns
string uniqueRegions = pandasManager.UniqueValues("customers", "Region");
Console.WriteLine($"\nRegions: {uniqueRegions}");

// Value distribution
string regionCounts = pandasManager.ValueCounts("customers", "Region");
Console.WriteLine("\nCustomers by Region:");
Console.WriteLine(regionCounts);

3. Basic Data Cleaning

// Remove rows with missing values
pandasManager.DropNA("sales", "sales_clean");

// Fill missing values with defaults
pandasManager.FillNA("customers", "customers_complete", "Unknown");

// Remove duplicates
pandasManager.DropDuplicates("products", "products_unique");

// Convert date columns
pandasManager.ConvertToDateTime("sales_clean", "OrderDate");

Console.WriteLine("Data cleaning completed!");

4. Simple Analysis

// Group sales by region

?? Common Usage Patterns

Pattern 1: ETL Pipeline

public async Task RunETLPipeline(string userId, string[] sourcePaths, string outputPath)
{
    // Configure session for user
    pandasManager.ConfigureSessionForUser(userId, "etl-environment");
    
    try
    {
        // Extract: Load from multiple sources
        for (int i = 0; i < sourcePaths.Length; i++)
        {
            await pandasManager.ReadCsvAsync($"source_{i}", sourcePaths[i]);
        }
        
        // Transform: Combine and clean data
        string[] sources = sourcePaths.Select((_, i) => $"source_{i}").ToArray();
        pandasManager.ConcatDataFrames(sources, "combined_data");
        
        pandasManager.DropNA("combined_data", "clean_data");
        pandasManager.DropDuplicates("clean_data", "final_data");
        
        // Load: Export to destination
        await pandasManager.ToCsvAsync("final_data", outputPath);
        
        Console.WriteLine($"ETL pipeline completed for user {userId}");
    }
    finally
    {
        // Cleanup resources
        pandasManager.CleanupIdleDataFrames();
    }
}

Pattern 2: Business Intelligence Dashboard

public class SalesDashboard
{
    private readonly IPythonPandasManager _pandasManager;
    
    public SalesDashboard(IPythonPandasManager pandasManager)
    {
        _pandasManager = pandasManager;
        _pandasManager.ConfigureSessionForUser("dashboard-service", "bi-environment");
    }
    
    public async Task<DashboardData> GenerateDashboard()
    {
        // Load sales data
        _pandasManager.ReadSql("sales", GetSalesQuery(), connectionString);
        
        // Create summary metrics
        _pandasManager.GroupBy("sales", "monthly_totals", "month", "sum");
        _pandasManager.GroupBy("sales", "regional_performance", "region", "sum");
        _pandasManager.PivotTable("sales", "product_matrix", "product", "month", "revenue");
        
        // Get statistical insights
        var totalRevenue = _pandasManager.CalculateMean("sales", "revenue");
        var topRegion = _pandasManager.ValueCounts("sales", "region");
        
        return new DashboardData
        {
            MonthlyTrends = _pandasManager.GetDataFrameAsJson("monthly_totals"),
            RegionalPerformance = _pandasManager.GetDataFrameAsJson("regional_performance"),
            ProductMatrix = _pandasManager.GetDataFrameAsJson("product_matrix"),
            AverageRevenue = totalRevenue,
            TopRegion = topRegion
        };
    }
}

Pattern 3: Data Validation Service

public class DataValidationService
{
    private readonly IPythonPandasManager _pandasManager;
    
    public async Task<ValidationResult> ValidateDataFile(string filePath, string userId)
    {
        _pandasManager.ConfigureSessionForUser(userId, "validation-env");
        
        var result = new ValidationResult();
        
        try
        {
            // Load data
            _pandasManager.ReadCsv("raw_data", filePath);
            
            // Check for issues
            var missingValues = _pandasManager.IsNull("raw_data");
            var duplicateCount = GetDuplicateCount("raw_data");
            var dataTypes = _pandasManager.Describe("raw_data");
            
            // Generate quality report
            result.HasMissingValues = !string.IsNullOrEmpty(missingValues);
            result.DuplicateRows = duplicateCount;
            result.Summary = dataTypes;
            result.IsValid = !result.HasMissingValues && result.DuplicateRows == 0;
            
            // Clean data if requested
            if (!result.IsValid)
            {
                pandasManager.DropNA("raw_data", "clean_data");
                pandasManager.DropDuplicates("clean_data", "validated_data");
                result.CleanedDataAvailable = true;
            }
        }
        catch (Exception ex)
        {
            result.ValidationError = ex.Message;
        }
        
        return result;
    }
}

?? Error Handling

Comprehensive Error Handling

public async Task<bool> SafeDataOperation(string dataPath)
{
    try
    {
        // Check session configuration
        if (!pandasManager.IsSessionConfigured())
        {
            Console.WriteLine("Configuring session...");
            if (!pandasManager.ConfigureSessionForUser("default-user"))
            {
                throw new InvalidOperationException("Failed to configure session");
            }
        }
        
        // Validate file exists
        if (!File.Exists(dataPath))
        {
            throw new FileNotFoundException($"Data file not found: {dataPath}");
        }
        
        // Load data with error handling
        Console.WriteLine("Loading data...");
        pandasManager.ReadCsv("data", dataPath);
        
        // Validate data was loaded
        string shape = pandasManager.GetDataFrameAsJson("data");
        if (string.IsNullOrEmpty(shape))
        {
            throw new InvalidOperationException("No data was loaded");
        }
        
        Console.WriteLine("Data loaded successfully");
        return true;
    }
    catch (FileNotFoundException ex)
    {
        Console.WriteLine($"File error: {ex.Message}");
        return false;
    }
    catch (InvalidOperationException ex)
    {
        Console.WriteLine($"Operation error: {ex.Message}");
        return false;
    }
    catch (Exception ex)
    {
        Console.WriteLine($"Unexpected error: {ex.Message}");
        return false;
    }
    finally
    {
        // Always cleanup resources
        try
        {
            pandasManager.CleanupIdleDataFrames();
        }
        catch (Exception cleanupEx)
        {
            Console.WriteLine($"Cleanup warning: {cleanupEx.Message}");
        }
    }
}

Common Issues and Solutions

  • Session not configured: Always check IsSessionConfigured() before operations
  • Python environment missing: Ensure Python and pandas are installed
  • File not found: Validate file paths before loading data
  • Memory issues: Use CleanupIdleDataFrames() regularly
  • Data type errors: Convert data types explicitly when needed

// Monitor memory usage
public void MonitorMemoryUsage()
{

    // Check memory again
    string updatedReport = pandasManager.GetMemoryUsageReport();
    Console.WriteLine("After Cleanup:");
    Console.WriteLine(updatedReport);
}
