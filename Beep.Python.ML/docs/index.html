?? Overview

Beep.Python.ML provides a comprehensive machine learning integration platform for .NET applications, enabling seamless interaction with Python ML frameworks like scikit-learn, TensorFlow, PyTorch, and XGBoost. The library offers enterprise-grade model management, training workflows, and evaluation capabilities.

Key Features
  • Multi-Framework Support: scikit-learn, TensorFlow, PyTorch, XGBoost, LightGBM
  • Model Management: Training, evaluation, persistence, and deployment
  • Algorithm Library: Comprehensive collection of ML algorithms and parameters
  • Evaluation Metrics: Built-in evaluation with visualization and reporting
  • MVVM Architecture: Complete ViewModels for UI integration
  • Hyperparameter Tuning: Grid search, random search, and Bayesian optimization
  • Model Visualization: Training curves, feature importance, and evaluation graphs

?? Quick Start

Basic ML Workflow

using Beep.Python.ML;
using TheTechIdea.Beep.Container.Services;

// Initialize the ML manager
var beepService = new BeepService();
var mlManager = new PythonMLManager(beepService);

// Initialize Python environment
await mlManager.InitializeAsync();

// Load and prepare data
await mlManager.LoadDataAsync("data.csv");
await mlManager.PrepareDataAsync(new DataPreparationOptions
{
    TargetColumn = "target",
    TestSize = 0.2,
    RandomState = 42,
    ScaleFeatures = true,
    HandleMissingValues = true
});

// Select and configure algorithm
var algorithm = mlManager.GetAlgorithm("RandomForestClassifier");
algorithm.SetParameters(new Dictionary<string, object>
{
    ["n_estimators"] = 100,
    ["max_depth"] = 10,
    ["random_state"] = 42
});

// Train model
var trainingResult = await mlManager.TrainModelAsync(algorithm);
if (trainingResult.Success)
{
    Console.WriteLine($"Model trained successfully in {trainingResult.TrainingTime}");
    Console.WriteLine($"Training Accuracy: {trainingResult.TrainingScore:F4}");
}

// Evaluate model
var evaluation = await mlManager.EvaluateModelAsync();
Console.WriteLine($"Test Accuracy: {evaluation.Accuracy:F4}");
Console.WriteLine($"Precision: {evaluation.Precision:F4}");
Console.WriteLine($"Recall: {evaluation.Recall:F4}");
Console.WriteLine($"F1-Score: {evaluation.F1Score:F4}");

// Make predictions
var predictions = await mlManager.PredictAsync(new[] { new double[] { 1.0, 2.0, 3.0 } });
Console.WriteLine($"Prediction: {predictions[0]}");

Advanced ML Workflow

// Advanced classification workflow
public async Task RunAdvancedClassificationWorkflow()
{
    var mlManager = new PythonMLManager(beepService);
    await mlManager.InitializeAsync();

    // Load and prepare data
    await mlManager.LoadDataAsync("customer_data.csv");
    await mlManager.PrepareDataAsync(new DataPreparationOptions
    {
        TargetColumn = "churn",
        TestSize = 0.3,
        RandomState = 42,
        ScaleFeatures = true,
        HandleMissingValues = true,
        MissingValueStrategy = MissingValueStrategy.Median
    });

    // Feature engineering
    await mlManager.EncodeCategoricalFeaturesAsync(EncodingMethod.OneHot);
    await mlManager.SelectFeaturesAsync(FeatureSelectionMethod.SelectKBest, 20);

    // Define algorithms to compare
    var algorithms = new[]
    {
        mlManager.GetAlgorithm("RandomForestClassifier"),
        mlManager.GetAlgorithm("GradientBoostingClassifier"),
        mlManager.GetAlgorithm("SupportVectorClassifier"),
        mlManager.GetAlgorithm("LogisticRegression")
    };

    // Train and compare models
    var trainingResults = await mlManager.TrainMultipleModelsAsync(algorithms);
    var comparisonResult = await mlManager.CompareModelsAsync(algorithms.Select(a => a.Name));

    // Select best model and tune hyperparameters
    var bestModel = comparisonResult.BestModel;
    var paramGrid = new Dictionary<string, object[]>
    {
        ["n_estimators"] = new object[] { 50, 100, 200 },
        ["max_depth"] = new object[] { 5, 10, 15, 20 },
        ["min_samples_split"] = new object[] { 2, 5, 10 }
    };

    var tuningResult = await mlManager.GridSearchAsync(bestModel, paramGrid);
    Console.WriteLine($"Best parameters: {string.Join(", ", tuningResult.BestParams)}");
    Console.WriteLine($"Best CV score: {tuningResult.BestScore:F4}");

    // Final evaluation
    var finalEvaluation = await mlManager.EvaluateModelAsync(tuningResult.BestModelName);
    await GenerateEvaluationReport(finalEvaluation);
}

// Comprehensive model evaluation
public async Task<EvaluationReport> GenerateEvaluationReport(string modelName)
{
    var evaluation = await mlManager.EvaluateModelAsync(modelName);
    
    var report = new EvaluationReport
    {
        ModelName = modelName,
        Timestamp = DateTime.UtcNow,
        
        // Classification Metrics
        Accuracy = evaluation.Accuracy,
        Precision = evaluation.Precision,
        Recall = evaluation.Recall,
        F1Score = evaluation.F1Score,
        AUC = evaluation.AUC,
        LogLoss = evaluation.LogLoss,
        
        // Confusion Matrix
        ConfusionMatrix = evaluation.ConfusionMatrix,
        ClassificationReport = evaluation.ClassificationReport,
        
        // Feature Importance
        FeatureImportance = await mlManager.GetFeatureImportanceAsync(modelName),
        
        // Learning Curves
        LearningCurves = await mlManager.GenerateLearningCurvesAsync(modelName),
        
        // ROC Curve
        ROCCurve = await mlManager.GenerateROCCurveAsync(modelName),
        
        // Precision-Recall Curve
        PrecisionRecallCurve = await mlManager.GeneratePrecisionRecallCurveAsync(modelName)
    };
    
    // Generate visualizations
    await GenerateEvaluationGraphs(report);
    
    return report;
}

public class PythonMachineLearningViewModel : PythonBaseViewModel
{
    // Core Properties
    public ObservableCollection<MLAlgorithmInfo> AvailableAlgorithms { get; set; }
    public MLAlgorithmInfo SelectedAlgorithm { get; set; }
    public ObservableCollection<ModelInfo> TrainedModels { get; set; }
    public ModelInfo SelectedModel { get; set; }
    
    // Data Properties
    public string DataFilePath { get; set; }
    public DataPreparationOptions DataPreparation { get; set; }
    public ObservableCollection<string> AvailableColumns { get; set; }
    public string TargetColumn { get; set; }
    
    // Training Properties
    public bool IsTraining { get; set; }
    public double TrainingProgress { get; set; }
    public string TrainingStatus { get; set; }
    public TrainingResult LastTrainingResult { get; set; }
    
    // Evaluation Properties
    public EvaluationResult CurrentEvaluation { get; set; }
    public ObservableCollection<EvaluationMetric> EvaluationMetrics { get; set; }
    
    // Commands
    public ICommand LoadDataCommand { get; set; }
    public ICommand PrepareDataCommand { get; set; }
    public ICommand TrainModelCommand { get; set; }
    public ICommand EvaluateModelCommand { get; set; }
    public ICommand SaveModelCommand { get; set; }
    public ICommand LoadModelCommand { get; set; }
    public ICommand PredictCommand { get; set; }
    public ICommand TuneHyperparametersCommand { get; set; }
    
    // Methods
    public async Task LoadDataAsync()
    {
        try
        {
            IsLoading = true;
            await MLManager.LoadDataAsync(DataFilePath);
            AvailableColumns = new ObservableCollection<string>(await MLManager.GetColumnNamesAsync());
            OnPropertyChanged(nameof(AvailableColumns));
        }
        catch (Exception ex)
        {
            ShowError($"Error loading data: {ex.Message}");
        }
        finally
        {
            IsLoading = false;
        }
    }
    
    public async Task TrainModelAsync()
    {
        try
        {
            IsTraining = true;
            TrainingProgress = 0;
            
            var progressReporter = new Progress<TrainingProgress>(progress =>
            {
                TrainingProgress = progress.Percentage;
                TrainingStatus = progress.Status;
            });
            
            LastTrainingResult = await MLManager.TrainModelAsync(SelectedAlgorithm, progressReporter);
            
            if (LastTrainingResult.Success)
            {
                TrainedModels.Add(new ModelInfo 
                { 
                    Name = SelectedAlgorithm.Name, 
                    TrainingScore = LastTrainingResult.TrainingScore,
                    TrainingTime = LastTrainingResult.TrainingTime
                });
                ShowSuccess("Model trained successfully!");
            }
        }
        catch (Exception ex)
        {
            ShowError($"Training failed: {ex.Message}");
        }
        finally
        {
            IsTraining = false;
            TrainingProgress = 0;
        }
    }
}

public class PythonTrainingViewModel : PythonBaseViewModel
{
    // Training Configuration
    public TrainingConfiguration Configuration { get; set; }
    public ObservableCollection<TrainingEpoch> TrainingHistory { get; set; }
    public LiveChartsCollection LearningCurveData { get; set; }
    
    // Hyperparameter Tuning
    public TuningStrategy SelectedTuningStrategy { get; set; }
    public Dictionary<string, ParameterRange> ParameterGrid { get; set; }
    public ObservableCollection<TuningResult> TuningResults { get; set; }
    
    // Cross-Validation
    public int CrossValidationFolds { get; set; }
    public CrossValidationResult CVResult { get; set; }
    
    // Advanced Training Features
    public bool UseEarlyStopping { get; set; }
    public int EarlyStoppingPatience { get; set; }
    public bool UseModelCheckpointing { get; set; }
    public string CheckpointDirectory { get; set; }
    
    // Real-time Training Monitoring
    public void OnTrainingEpochCompleted(TrainingEpoch epoch)
    {
        TrainingHistory.Add(epoch);
        UpdateLearningCurves(epoch);
        
        if (UseEarlyStopping && ShouldStopEarly(epoch))
        {
            RequestTrainingStop();
        }
    }
    
    private void UpdateLearningCurves(TrainingEpoch epoch)
    {
        LearningCurveData.Add(new ChartPoint(epoch.EpochNumber, epoch.TrainingLoss, "Training Loss"));
        LearningCurveData.Add(new ChartPoint(epoch.EpochNumber, epoch.ValidationLoss, "Validation Loss"));
        LearningCurveData.Add(new ChartPoint(epoch.EpochNumber, epoch.TrainingAccuracy, "Training Accuracy"));
        LearningCurveData.Add(new ChartPoint(epoch.EpochNumber, epoch.ValidationAccuracy, "Validation Accuracy"));
    }
}

? Best Practices

ML Development Best Practices
  • Data Quality: Always validate and clean your data before training
  • Cross-Validation: Use proper cross-validation to assess model performance
  • Feature Engineering: Invest time in creating meaningful features
  • Model Selection: Try multiple algorithms and compare their performance
  • Hyperparameter Tuning: Systematically optimize model hyperparameters
  • Model Monitoring: Monitor model performance in production
Common Pitfalls
  • Data Leakage: Ensure no future information leaks into training data
  • Overfitting: Monitor validation performance to detect overfitting
  • Underfitting: Ensure model complexity matches problem complexity
  • Imbalanced Data: Address class imbalance appropriately
  • Feature Scaling: Scale features when using distance-based algorithms

public async Task CustomerChurnPrediction()
{
    var mlManager = new PythonMLManager(beepService);
    await mlManager.InitializeAsync();

    // Load customer data
    await mlManager.LoadDataAsync("customer_churn.csv");
    
    // Data preparation
    await mlManager.PrepareDataAsync(new DataPreparationOptions
    {
        TargetColumn = "Churn",
        TestSize = 0.3,
        RandomState = 42,
        HandleMissingValues = true,
        ScaleFeatures = true
    });

    // Feature engineering
    await mlManager.EncodeCategoricalFeaturesAsync(EncodingMethod.OneHot);
    
    // Train multiple models
    var algorithms = new[]
    {
        mlManager.GetAlgorithm("RandomForestClassifier"),
        mlManager.GetAlgorithm("XGBClassifier"),
        mlManager.GetAlgorithm("LogisticRegression")
    };

    var results = await mlManager.TrainMultipleModelsAsync(algorithms);
    var bestModel = results.OrderByDescending(r => r.ValidationScore).First();

    // Hyperparameter tuning
    var paramGrid = new Dictionary<string, object[]>
    {
        ["n_estimators"] = new object[] { 100, 200, 300 },
        ["max_depth"] = new object[] { 5, 10, 15 },
        ["learning_rate"] = new object[] { 0.1, 0.2, 0.3 }
    };

    var tuningResult = await mlManager.GridSearchAsync(bestModel.Algorithm, paramGrid);
    
    // Final evaluation
    var evaluation = await mlManager.EvaluateModelAsync(tuningResult.BestModelName);
    
    Console.WriteLine($"Final Model Performance:");
    Console.WriteLine($"Accuracy: {evaluation.Accuracy:F4}");
    Console.WriteLine($"Precision: {evaluation.Precision:F4}");
    Console.WriteLine($"Recall: {evaluation.Recall:F4}");
    Console.WriteLine($"F1-Score: {evaluation.F1Score:F4}");
    Console.WriteLine($"AUC: {evaluation.AUC:F4}");
}
