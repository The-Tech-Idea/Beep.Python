private void OnAutoMLProgressChanged(AutoMLProgress progress)
{
    Console.WriteLine($"[{progress.Percentage:F1}%] {progress.Stage}: {progress.Message}");
    
    if (progress.CurrentTrial != null)
    {
        Console.WriteLine($"  Trial {progress.CurrentTrial.Number}: {progress.CurrentTrial.Algorithm} " +
                         $"Score: {progress.CurrentTrial.Score:F4}");
    }
    
    if (progress.BestTrialSoFar != null)
    {
        Console.WriteLine($"  Best so far: {progress.BestTrialSoFar.Algorithm} " +
                         $"Score: {progress.BestTrialSoFar.Score:F4}");
    }
}

public class AutoMLConfiguration

    // Time and resource constraints

    public string OptimizationMetric { get; set; } = "f1_weighted";

    public int EarlyStoppingRounds { get; set; } = 10;
    public double EarlyStoppingThreshold { get; set; } = 0.001;
    
    // Feature engineering
    public bool EnableFeatureEngineering { get; set; } = true;
    public bool EnableAutomaticFeatureSelection { get; set; } = true;
    public int MaxFeatures { get; set; } = 50;
    public string FeatureSelectionMethod { get; set; } = "mutual_info";

    public int PolynomialDegree { get; set; } = 2;

    public MachineLearningAlgorithm[] IncludeAlgorithms { get; set; }

    public bool EnableStackedEnsemble { get; set; } = false;
    
    // Cross-validation

public async Task<AutoMLResult> RunAutoMLAsync(
    AutoMLConfiguration config, 
    IProgress<AutoMLProgress> progress = null)
{
    var result = new AutoMLResult { StartTime = DateTime.Now };
    var totalStages = 8;
    var currentStage = 0;
    
    try
    {
        // Stage 1: Data Analysis and Validation
        currentStage++;
        progress?.Report(new AutoMLProgress
        {
            Stage = "Data Analysis",
            Percentage = (currentStage * 100.0 / totalStages),
            Message = "Analyzing data characteristics and quality"
        });
        
        var dataAnalysis = await AnalyzeDataAsync();
        if (!dataAnalysis.IsValid)
        {
            result.ErrorMessage = $"Data validation failed: {dataAnalysis.ErrorMessage}";
            return result;
        }
        
        // Stage 2: Automatic Data Preprocessing
        currentStage++;
        progress?.Report(new AutoMLProgress
        {
            Stage = "Data Preprocessing",
            Percentage = (currentStage * 100.0 / totalStages),
            Message = "Applying automatic data preprocessing"
        });
        
        await ApplyAutomaticPreprocessingAsync(config);
        
        // Stage 3: Feature Engineering
        if (config.EnableFeatureEngineering)
        {
            currentStage++;
            progress?.Report(new AutoMLProgress
            {
                Stage = "Feature Engineering",
                Percentage = (currentStage * 100.0 / totalStages),
                Message = "Generating and selecting features automatically"
            });
            
            var featureReport = await PerformAutomaticFeatureEngineeringAsync(config);
            result.FeatureEngineeringReport = featureReport;
        }
        
        // Stage 4: Algorithm Selection and Training
        currentStage++;
        progress?.Report(new AutoMLProgress
        {
            Stage = "Model Selection",
            Percentage = (currentStage * 100.0 / totalStages),
            Message = "Testing multiple algorithms and configurations"
        });
        
        var trials = await RunAlgorithmTrialsAsync(config, progress);
        result.AllTrials = trials;
        
        // Find best trial
        var bestTrial = trials.OrderByDescending(t => t.Score).First();
        result.BestAlgorithm = bestTrial.Algorithm;
        result.BestParameters = bestTrial.Parameters;
        result.BestScore = bestTrial.Score;
        
        // Stage 5: Hyperparameter Optimization for Best Model
        currentStage++;
        progress?.Report(new AutoMLProgress
        {
            Stage = "Hyperparameter Optimization",
            Percentage = (currentStage * 100.0 / totalStages),
            Message = $"Optimizing {bestTrial.Algorithm} hyperparameters"
        });
        
        var optimizedResult = await OptimizeBestModelAsync(bestTrial, config);
        if (optimizedResult.Success)
        {
            result.BestParameters = optimizedResult.BestParams;
            result.BestScore = optimizedResult.BestScore;
        }
        
        // Stage 6: Ensemble Creation
        if (config.EnableEnsemble)
        {
            currentStage++;
            progress?.Report(new AutoMLProgress
            {
                Stage = "Ensemble Creation",
                Percentage = (currentStage * 100.0 / totalStages),
                Message = "Creating ensemble from top models"
            });
            
            var topTrials = trials.OrderByDescending(t => t.Score).Take(config.EnsembleSize);
            var ensembleResult = await CreateEnsembleFromTrialsAsync(topTrials, config);
            
            if (ensembleResult.Success && ensembleResult.EnsembleScore > result.BestScore)
            {
                result.BestScore = ensembleResult.EnsembleScore;
                result.EnsembleResult = ensembleResult;
                result.UsedEnsemble = true;
                result.BestAlgorithm = MachineLearningAlgorithm.EnsembleClassifier;
            }
        }
        
        // Stage 7: Final Model Training and Validation
        currentStage++;
        progress?.Report(new AutoMLProgress
        {
            Stage = "Final Training",
            Percentage = (currentStage * 100.0 / totalStages),
            Message = "Training final model with best configuration"
        });
        
        var finalModelId = await TrainFinalModelAsync(result);
        result.BestModelId = finalModelId;
        
        // Stage 8: Model Evaluation and Report Generation
        currentStage++;
        progress?.Report(new AutoMLProgress
        {
            Stage = "Evaluation & Reporting",
            Percentage = 100,
            Message = "Generating comprehensive evaluation report"
        });
        
        result.BestModelMetrics = await EvaluateFinalModelAsync(finalModelId);
        
        if (config.GenerateReport)
        {
            var reportPath = await GenerateAutoMLReportAsync(result, config);
            result.ReportFilePath = reportPath;
        }
        
        // Save model if requested
        if (config.SaveIntermediateModels)
        {
            var modelPath = await SaveFinalModelAsync(finalModelId, config.OutputDirectory);
            result.ModelFilePath = modelPath;
        }
        
        result.Success = true;
        result.EndTime = DateTime.Now;
        result.TotalRunTime = result.EndTime - result.StartTime;
        result.CompletedTrials = trials.Count;
        
        progress?.Report(new AutoMLProgress
        {
            Stage = "Complete",
            Percentage = 100,
            Message = $"AutoML completed successfully! Best score: {result.BestScore:F4}"
        });
        
    }
    catch (Exception ex)
    {
        result.ErrorMessage = ex.Message;
        result.Success = false;
    }
    
    return result;
}

// AutoML integration in PythonMachineLearningViewModel
public partial class PythonMachineLearningViewModel : PythonBaseViewModel
{
    // AutoML properties
    [ObservableProperty] private AutoMLConfiguration autoMLConfig;
    [ObservableProperty] private bool isAutoMLRunning;
    [ObservableProperty] private AutoMLProgress currentAutoMLProgress;
    [ObservableProperty] private AutoMLResult autoMLResult;
    [ObservableProperty] private ObservableCollection<TrialResult> autoMLTrials;
    
    // Commands
    public IAsyncRelayCommand RunAutoMLCommand { get; }
    public IAsyncRelayCommand StopAutoMLCommand { get; }
    public IRelayCommand ConfigureAutoMLCommand { get; }
    
    public PythonMachineLearningViewModel(IBeepService beepService, 
                                        IPythonNetRunTimeManager pythonRuntime, 
                                        PythonSessionInfo sessionInfo)
        : base(beepService, pythonRuntime, sessionInfo)
    {
        // Initialize AutoML properties
        AutoMLConfig = new AutoMLConfiguration();
        AutoMLTrials = new ObservableCollection<TrialResult>();
        
        // Initialize commands
        RunAutoMLCommand = new AsyncRelayCommand(RunAutoMLAsync, CanRunAutoML);
        StopAutoMLCommand = new AsyncRelayCommand(StopAutoMLAsync, () => IsAutoMLRunning);
        ConfigureAutoMLCommand = new RelayCommand(ConfigureAutoML);
    }
    
    private async Task RunAutoMLAsync()
    {
        try
        {
            IsAutoMLRunning = true;
            CurrentWorkflowStage = "Starting AutoML...";
            
            // Progress reporting
            var progress = new Progress<AutoMLProgress>(OnAutoMLProgressChanged);
            
            // Run AutoML
            var result = await RunAutoMLAsync(AutoMLConfig, progress);
            
            if (result.Success)
            {
                AutoMLResult = result;
                
                // Update UI with results
                SelectedAlgorithmInfo = new MLAlgorithmInfo
                {
                    Algorithm = result.BestAlgorithm,
                    Name = result.BestAlgorithm.ToString(),
                    Parameters = result.BestParameters
                };
                
                CurrentModelMetrics = result.BestModelMetrics;
                
                // Update trials display
                AutoMLTrials.Clear();
                foreach (var trial in result.AllTrials.OrderByDescending(t => t.Score))
                {
                    AutoMLTrials.Add(trial);
                }
                
                CurrentWorkflowStage = $"AutoML Completed - Best: {result.BestAlgorithm} ({result.BestScore:F4})";
                ShowSuccessMessage($"AutoML found optimal model with {result.OptimizationMetric}: {result.BestScore:F4}");
            }
            else
            {
                ShowErrorMessage($"AutoML failed: {result.ErrorMessage}");
            }
        }
        catch (Exception ex)
        {
            ShowErrorMessage($"AutoML execution error: {ex.Message}");
        }
        finally
        {
            IsAutoMLRunning = false;
        }
    }
    
    private void OnAutoMLProgressChanged(AutoMLProgress progress)
    {
        CurrentAutoMLProgress = progress;
        OverallProgress = progress.Percentage;
        CurrentWorkflowStage = $"{progress.Stage}: {progress.Message}";
        
        // Update trials in real-time
        if (progress.CurrentTrial != null)
        {
            Dispatcher.BeginInvoke(() =>
            {
                // Add or update trial in collection
                var existingTrial = AutoMLTrials.FirstOrDefault(t => t.Number == progress.CurrentTrial.Number);
                if (existingTrial != null)
                {
                    var index = AutoMLTrials.IndexOf(existingTrial);
                    AutoMLTrials[index] = progress.CurrentTrial;
                }
                else
                {
                    AutoMLTrials.Add(progress.CurrentTrial);
                }
            });
        }
        
        // Update best trial display
        if (progress.BestTrialSoFar != null)
        {
            NotifyBestTrialUpdate(progress.BestTrialSoFar);
        }
    }
    
    private bool CanRunAutoML()
    {
        return !IsAutoMLRunning && 
               IsMLPipelineReady && 
               !string.IsNullOrEmpty(LabelColumn) && 
               Features?.Any() == true;
    }
}

? AutoML Best Practices

AutoML Optimization Guidelines
  • Data Quality: Ensure clean, representative data for best results
  • Time Budget: Allocate sufficient time for thorough search
  • Metric Selection: Choose metrics aligned with business objectives
  • Feature Engineering: Enable for better automated feature discovery
  • Cross-Validation: Use appropriate CV strategy for your data
  • Baseline Comparison: Compare AutoML results with simple baselines
AutoML Limitations
  • Domain Knowledge: AutoML can't replace domain expertise
  • Data Understanding: Still need to understand your data
  • Computational Cost: Can be resource-intensive
  • Interpretability: Complex models may be hard to explain
  • Overfitting: Risk of overfitting to validation set
