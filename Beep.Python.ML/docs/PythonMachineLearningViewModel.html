    public int MaxTrials { get; set; } = 100;
    public string OptimizationMetric { get; set; } = "f1_weighted";
    public bool EnableFeatureEngineering { get; set; } = true;
    public bool EnableEnsemble { get; set; } = true;
    public bool EnableStackedEnsemble { get; set; } = false;

// AutoML command implementation
private async Task RunAutoMLAsync()
{
    try
    {
        IsAutoMLRunning = true;
        CurrentWorkflowStage = "AutoML Initialization";
        
        // Configure AutoML settings
        var autoMLConfig = new AutoMLConfiguration
        {
            MaxRunTime = TimeSpan.FromHours(2),
            MaxTrials = AutoMLConfig?.MaxTrials ?? 100,
            OptimizationMetric = "f1_weighted",
            EnableFeatureEngineering = true,
            EnableEnsemble = true
        };
        
        // Progress reporting
        var progress = new Progress<AutoMLProgress>(OnAutoMLProgressChanged);
        
        // Execute AutoML workflow
        var result = await ExecuteAutoMLWorkflowAsync(autoMLConfig, progress);
        
        if (result.Success)
        {
            CurrentAutoMLResult = result;
            CurrentModelMetrics = result.BestModelMetrics;
            
            // Update UI with best model
            SelectedAlgorithmInfo = result.BestAlgorithm;
            
            CurrentWorkflowStage = $"AutoML Completed - Best Score: {result.BestScore:F4}";
            ShowSuccessMessage($"AutoML found optimal model: {result.BestAlgorithm.Name} with {result.OptimizationMetric}: {result.BestScore:F4}");
        }
    }
    catch (Exception ex)
    {
        ShowErrorMessage($"AutoML failed: {ex.Message}");
    }
    finally
    {
        IsAutoMLRunning = false;
    }
}

// Create ensemble command
private async Task CreateEnsembleAsync()
{
    try
    {
        CurrentWorkflowStage = "Creating Ensemble";
        
        // Define base models for ensemble
        var baseAlgorithms = new[]
        {
            MachineLearningAlgorithm.RandomForestClassifier,
            MachineLearningAlgorithm.GradientBoostingClassifier,
            MachineLearningAlgorithm.LogisticRegression,
            MachineLearningAlgorithm.SVC
        };
        
        // Configure ensemble
        var ensembleConfig = new EnsembleConfiguration
        {
            Type = EnsembleType.Stacking,
            VotingType = VotingType.Soft,
            UseStackedEnsemble = true,
            MetaLearner = MachineLearningAlgorithm.LogisticRegression,
            BaseAlgorithms = baseAlgorithms.ToList()
        };
        
        // Execute ensemble training using TrainingExtensions
        var result = await this.TrainEnsembleAsync(
            ensembleConfig.Type, 
            baseAlgorithms);
        
        if (result.Success)
        {
            CurrentEnsembleResult = result;
            CurrentModelMetrics = new ModelPerformanceMetrics
            {
                Accuracy = result.EnsembleScore,
                TrainingTime = result.TrainingTime,
                ModelType = "Ensemble",
                EstimatorCount = result.EstimatorCount
            };
            
            CurrentWorkflowStage = $"Ensemble Created - Score: {result.EnsembleScore:F4}";
            ShowSuccessMessage($"Ensemble model created with score: {result.EnsembleScore:F4}");
        }
    }
    catch (Exception ex)
    {
        ShowErrorMessage($"Ensemble creation failed: {ex.Message}");
    }

public class HyperparameterOptimizationConfig
{
    public SearchType SearchType { get; set; } = SearchType.BayesianOptimization;
    public int MaxIterations { get; set; } = 100;
    public int CVFolds { get; set; } = 5;
    public string ScoringMetric { get; set; } = "f1_weighted";
    public Dictionary<string, object[]> ParameterGrid { get; set; } = new();
    public bool UseEarlyStopping { get; set; } = true;
    public double EarlyStoppingRounds { get; set; } = 10;
}

// Optimize hyperparameters command
private async Task OptimizeHyperparametersAsync()
{
    try
    {
        IsOptimizationRunning = true;
        CurrentWorkflowStage = "Optimizing Hyperparameters";
        
        // Configure optimization
        var config = new HyperparameterOptimizationConfig
        {
            SearchType = SearchType.BayesianOptimization,
            MaxIterations = OptimizationConfig?.MaxIterations ?? 100,
            CVFolds = 5,
            ScoringMetric = "f1_weighted",
            UseEarlyStopping = true
        };
        
        // Define parameter grid based on selected algorithm
        config.ParameterGrid = GetParameterGridForAlgorithm(SelectedAlgorithmInfo.Algorithm);
        
        // Progress reporting
        var progress = new Progress<OptimizationProgress>(OnOptimizationProgressChanged);
        
        // Execute optimization using TrainingExtensions
        var result = await this.OptimizeHyperparametersAdvancedAsync(
            config.SearchType,
            config.ParameterGrid,
            config.MaxIterations,
            progress);
        
        if (result.Success)
        {
            OptimizationResult = result;
            
            // Update model with best parameters
            UpdateModelWithOptimizedParameters(result.BestParams);
            
            CurrentWorkflowStage = $"Optimization Complete - Best Score: {result.BestScore:F4}";
            ShowSuccessMessage($"Hyperparameter optimization completed. Best score: {result.BestScore:F4}");
        }
    }
    catch (Exception ex)
    {
        ShowErrorMessage($"Hyperparameter optimization failed: {ex.Message}");
    }
    finally
    {
        IsOptimizationRunning = false;
    }
}

private Dictionary<string, object[]> GetParameterGridForAlgorithm(MachineLearningAlgorithm algorithm)
{
    return algorithm switch
    {
        MachineLearningAlgorithm.RandomForestClassifier => new Dictionary<string, object[]>
        {
            ["n_estimators"] = new object[] { 50, 100, 200, 300 },
            ["max_depth"] = new object[] { 5, 10, 15, 20, null },
            ["min_samples_split"] = new object[] { 2, 5, 10 },
            ["min_samples_leaf"] = new object[] { 1, 2, 4 },
            ["max_features"] = new object[] { "sqrt", "log2", "auto" }
        },
        MachineLearningAlgorithm.GradientBoostingClassifier => new Dictionary<string, object[]>
        {
            ["n_estimators"] = new object[] { 100, 200, 300 },
            ["learning_rate"] = new object[] { 0.01, 0.1, 0.2 },
            ["max_depth"] = new object[] { 3, 5, 7, 10 },
            ["subsample"] = new object[] { 0.8, 0.9, 1.0 }
        },
        MachineLearningAlgorithm.SVC => new Dictionary<string, object[]>
        {
            ["C"] = new object[] { 0.1, 1, 10, 100 },
            ["kernel"] = new object[] { "linear", "rbf", "poly" },
            ["gamma"] = new object[] { "scale", "auto", 0.001, 0.01, 0.1 }
        },
        _ => new Dictionary<string, object[]> { ["random_state"] = new object[] { 42 } }
    };
}

// Comprehensive preprocessing workflow using all assistants
public async Task ExecuteComprehensivePreprocessingAsync()
{
    try
    {
        CurrentWorkflowStage = "Comprehensive Preprocessing";
        
        if (MLManager is not PythonMLManager manager)
        {
            throw new InvalidOperationException("PythonMLManager required for assistant access");
        }
        
        // Phase 1: Data Cleaning
        OverallProgress = 10;
        CurrentWorkflowStage = "Phase 1: Data Cleaning";
        
        manager.DataCleaning.RemoveOutliers(zThreshold: 3.0);
        manager.DataCleaning.RemoveDuplicates();
        manager.DataCleaning.StandardizeCategories();
        
        // Phase 2: Missing Value Handling
        OverallProgress = 20;
        CurrentWorkflowStage = "Phase 2: Missing Value Handling";
        
        manager.DataPreprocessing.ImputeMissingValues("median");
        manager.DataPreprocessing.HandleMissingValuesAdvanced("knn", neighbors: 5);
        
        // Phase 3: Feature Engineering
        OverallProgress = 35;
        CurrentWorkflowStage = "Phase 3: Feature Engineering";
        
        manager.FeatureEngineering.GeneratePolynomialFeatures(degree: 2, interactionOnly: true);
        manager.FeatureEngineering.ApplyLogTransformation(new[] { "revenue", "income" });
        manager.FeatureEngineering.CreateBinningFeatures("age", bins: 5, strategy: "quantile");
        
        // Phase 4: Categorical Processing
        OverallProgress = 50;
        CurrentWorkflowStage = "Phase 4: Categorical Processing";
        
        var categoricalFeatures = manager.CategoricalEncoding.GetCategoricalFeatures(Features);
        manager.CategoricalEncoding.TargetEncode(categoricalFeatures, LabelColumn);
        
        // Phase 5: Text Processing (if applicable)
        OverallProgress = 65;
        CurrentWorkflowStage = "Phase 5: Text Processing";
        
        var textColumns = GetTextColumns();
        if (textColumns.Any())
        {
            foreach (var column in textColumns)
            {
                manager.TextProcessing.ConvertTextToLowercase(column);
                manager.TextProcessing.RemoveStopwords(column, "english");
                manager.TextProcessing.ApplyTFIDFVectorization(column, maxFeatures: 1000);
            }
        }
        
        // Phase 6: DateTime Processing
        OverallProgress = 75;
        CurrentWorkflowStage = "Phase 6: DateTime Processing";
        
        var dateTimeColumns = GetDateTimeColumns();
        foreach (var column in dateTimeColumns)
        {
            manager.DateTimeProcessing.ExtractDateTimeComponents(column);
            manager.DateTimeProcessing.HandleCyclicalTimeFeatures(column, "cyclical");
        }
        
        // Phase 7: Feature Selection
        OverallProgress = 85;
        CurrentWorkflowStage = "Phase 7: Feature Selection";
        
        manager.FeatureSelection.ApplyVarianceThreshold(threshold: 0.01);
        manager.FeatureSelection.ApplyCorrelationThreshold(threshold: 0.95);
        manager.FeatureSelection.ApplyRFE("RandomForestRegressor", nFeatures: 50);
        
        // Phase 8: Data Balancing
        OverallProgress = 95;
        CurrentWorkflowStage = "Phase 8: Data Balancing";
        
        manager.ImbalancedData.ApplySMOTE(LabelColumn, samplingStrategy: 1.0f);
        
        // Phase 9: Final Scaling and Validation
        OverallProgress = 100;
        CurrentWorkflowStage = "Phase 9: Final Processing";
        
        manager.DataPreprocessing.RobustScaleData();
        
        CurrentWorkflowStage = "Comprehensive Preprocessing Completed";
        ShowSuccessMessage("All 13 assistant classes applied successfully!");
        
    }
    catch (Exception ex)
    {
        ShowErrorMessage($"Comprehensive preprocessing failed: {ex.Message}");
        CurrentWorkflowStage = "Preprocessing Failed";
    }
}

// Advanced workflow configuration
[ObservableProperty] private AdvancedWorkflowConfiguration workflowConfig;

public class AdvancedWorkflowConfiguration
{
    public bool EnableAutoML { get; set; } = false;
    public bool EnableEnsemble { get; set; } = true;
    public bool EnableHyperparameterOptimization { get; set; } = true;
    public bool EnableCrossValidation { get; set; } = true;
    public bool EnableComprehensiveEvaluation { get; set; } = true;
    public bool GenerateReport { get; set; } = true;
    public bool SaveModels { get; set; } = true;
    public string OutputDirectory { get; set; } = "outputs";
}

// Start advanced workflow command
private async Task StartAdvancedWorkflowAsync()
{
    try
    {
        IsAdvancedWorkflowRunning = true;
        OverallProgress = 0;
        CurrentWorkflowStage = "Initializing Advanced Workflow";
        
        // Use TrainingExtensions for comprehensive workflow
        var config = new AdvancedTrainingConfiguration
        {
            EnablePreprocessing = true,
            EnableHyperparameterOptimization = WorkflowConfig?.EnableHyperparameterOptimization ?? true,
            EnableCrossValidation = WorkflowConfig?.EnableCrossValidation ?? true,
            EnableEnsemble = WorkflowConfig?.EnableEnsemble ?? true,
            GenerateReport = WorkflowConfig?.GenerateReport ?? true,
            SaveModel = WorkflowConfig?.SaveModels ?? true,
            OutputDirectory = WorkflowConfig?.OutputDirectory ?? "outputs"
        };
        
        // Progress reporting
        var progress = new Progress<MLTrainingProgress>(OnWorkflowProgressChanged);
        
        // Execute comprehensive training workflow
        var result = await this.ExecuteAdvancedTrainingAsync(config, progress);
        
        if (result.Success)
        {
            WorkflowResult = result;
            CurrentModelMetrics = result.FinalMetrics;
            
            // Update training history
            TrainingHistory.Add(new TrainingHistoryItem
            {
                Timestamp = result.EndTime,
                Algorithm = result.BestAlgorithm,
                Score = result.FinalMetrics.Accuracy,
                Duration = result.Duration,
                Configuration = config
            });
            
            CurrentWorkflowStage = "Advanced Workflow Completed Successfully";
            ShowSuccessMessage($"Advanced ML workflow completed in {result.Duration.TotalMinutes:F1} minutes!");
            
            // Generate comprehensive report if enabled
            if (config.GenerateReport)
            {
                await GenerateComprehensiveReportAsync(result);
            }
        }
        else
        {
            ShowErrorMessage($"Advanced workflow failed: {result.ErrorMessage}");
        }
    }
    catch (Exception ex)
    {
        ShowErrorMessage($"Advanced workflow execution failed: {ex.Message}");
    }
    finally
    {
        IsAdvancedWorkflowRunning = false;
        OverallProgress = 0;
    }
}

// Progress event handlers
private void OnWorkflowProgressChanged(MLTrainingProgress progress)
{
    CurrentWorkflowStage = progress.Message;
    OverallProgress = progress.Progress;
    
    // Update additional progress information
    if (progress.EstimatedRemaining.HasValue)
    {
        // Update estimated time remaining in UI
        NotifyEstimatedTimeRemaining(progress.EstimatedRemaining.Value);
    }
}

private void OnAutoMLProgressChanged(AutoMLProgress progress)
{
    CurrentWorkflowStage = $"AutoML: {progress.Message}";
    OverallProgress = progress.Percentage;
    
    // Update current trial information
    if (progress.CurrentTrialInfo != null)
    {
        // Update UI with current trial details
        NotifyCurrentTrialInfo(progress.CurrentTrialInfo);
    }
}

private void OnOptimizationProgressChanged(OptimizationProgress progress)
{
    CurrentWorkflowStage = $"Optimization: {progress.Message}";
    OverallProgress = progress.Percentage;
    
    // Update best score so far
    if (progress.BestScoreSoFar.HasValue)
    {
        NotifyBestScoreUpdate(progress.BestScoreSoFar.Value);
    }
}

    public double F1Score { get; set; }

    public string ModelType { get; set; }
    public int EstimatorCount { get; set; }
    public Dictionary<string, object> Hyperparameters { get; set; }
}

public class PerformanceAnalytics
{
    public double AverageAccuracy { get; set; }
    public double BestAccuracy { get; set; }
    public string BestPerformingAlgorithm { get; set; }
    public TimeSpan AverageTrainingTime { get; set; }
    public int TotalModelsTrained { get; set; }
    public Dictionary<string, double> AlgorithmPerformanceComparison { get; set; }
    public List<PerformanceTrend> PerformanceTrends { get; set; }
}

// Performance monitoring methods
public void UpdatePerformanceMetrics(ModelPerformanceMetrics metrics)
{
    PerformanceHistory.Add(metrics);
    CurrentModelMetrics = metrics;
    
    // Update analytics
    RecalculateAnalytics();
    
    // Trigger performance analysis
    AnalyzePerformanceTrends();
}

private void RecalculateAnalytics()
{
    if (!PerformanceHistory.Any()) return;
    
    CurrentAnalytics = new PerformanceAnalytics
    {
        AverageAccuracy = PerformanceHistory.Average(p => p.Accuracy),
        BestAccuracy = PerformanceHistory.Max(p => p.Accuracy),
        BestPerformingAlgorithm = PerformanceHistory
            .OrderByDescending(p => p.Accuracy)
            .First().Algorithm,
        AverageTrainingTime = TimeSpan.FromTicks(
            (long)PerformanceHistory.Average(p => p.TrainingTime.Ticks)),
        TotalModelsTrained = PerformanceHistory.Count,
        AlgorithmPerformanceComparison = PerformanceHistory
            .GroupBy(p => p.Algorithm)
            .ToDictionary(g => g.Key, g => g.Average(p => p.Accuracy))
    };
}

// Generate comprehensive report
private async Task GenerateComprehensiveReportAsync(ComprehensiveTrainingResult result)
{
    try
    {
        CurrentWorkflowStage = "Generating Comprehensive Report";
        
        // Generate advanced report using TrainingExtensions
        var report = this.GenerateAdvancedReport(includeCharts: true);
        
        // Save report to file
        var reportPath = Path.Combine(result.Configuration.OutputDirectory, "comprehensive_report.md");
        await File.WriteAllTextAsync(reportPath, report);
        
        // Generate HTML version if visualization assistant is available
        if (MLManager is PythonMLManager manager)
        {
            var htmlReportPath = Path.Combine(result.Configuration.OutputDirectory, "comprehensive_report.html");
            manager.Visualization.GenerateEvaluationReport(result.BestModelId, htmlReportPath);
        }
        
        ShowSuccessMessage($"Comprehensive report generated: {reportPath}");
    }
    catch (Exception ex)
    {
        ShowErrorMessage($"Report generation failed: {ex.Message}");
    }
}
