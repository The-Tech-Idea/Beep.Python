// Generate comprehensive performance charts
private async Task GeneratePerformanceChartsAsync()
{
    try
    {
        IsGeneratingCharts = true;
        CurrentChartType = "Performance Metrics";
        ChartGenerationProgress = 0;
        
        // Progress tracking
        var progress = new Progress<ChartGenerationProgress>(OnChartProgressChanged);
        
        // Generate multiple performance charts
        await GenerateAccuracyTrendsChart(progress);
        await GenerateLossEvolutionChart(progress);
        await GenerateMetricsComparisonChart(progress);
        await GenerateTrainingValidationChart(progress);
        
        // Update chart data collections
        await UpdatePerformanceDataCollection();
        
        CurrentChartType = "Performance Charts Generated";
        IsVisualizationReady = true;
        
        ShowSuccessMessage("Performance charts generated successfully!");
    }
    catch (Exception ex)
    {
        ShowErrorMessage($"Chart generation failed: {ex.Message}");
    }
    finally
    {
        IsGeneratingCharts = false;
        ChartGenerationProgress = 100;
    }
}

private async Task GenerateAccuracyTrendsChart(IProgress<ChartGenerationProgress> progress)
{
    progress?.Report(new ChartGenerationProgress 
    { 
        Stage = "Accuracy Trends", 
        Percentage = 25, 
        Message = "Generating accuracy trends visualization" 
    });
    
    // Use PythonVisualizationAssistant for chart generation
    if (MLManager is PythonMLManager manager)
    {
        var chartData = await manager.Visualization.GenerateAccuracyTrendsChart(
            modelId: EvaluationResults?.ModelId,
            outputPath: GetChartOutputPath("accuracy_trends.png"),
            showValidation: true,
            theme: CurrentTheme.ToString().ToLower());
        
        // Update observable collection
        var dataPoints = ConvertToChartDataPoints(chartData, "Accuracy Trends");
        PerformanceData.Clear();
        foreach (var point in dataPoints)
        {
            PerformanceData.Add(point);
        }
    }
}

// Generate interactive confusion matrix
private async Task GenerateConfusionMatrixAsync()
{
    try
    {
        CurrentChartType = "Confusion Matrix";
        
        if (MLManager is PythonMLManager manager && EvaluationResults != null)
        {
            // Generate confusion matrix data
            var matrixData = await manager.Visualization.GenerateConfusionMatrix(
                modelId: EvaluationResults.ModelId,
                normalize: true,
                includePercentages: true,
                colorScheme: CurrentTheme.ToString());
            
            // Convert to observable collection
            ConfusionMatrixData.Clear();
            
            for (int i = 0; i < matrixData.ActualLabels.Length; i++)
            {
                for (int j = 0; j < matrixData.PredictedLabels.Length; j++)
                {
                    ConfusionMatrixData.Add(new ConfusionMatrixData
                    {
                        ActualClass = matrixData.ActualLabels[i],
                        PredictedClass = matrixData.PredictedLabels[j],
                        Count = matrixData.Matrix[i, j],
                        Percentage = matrixData.NormalizedMatrix[i, j],
                        IsCorrectPrediction = i == j,
                        CellColor = GetConfusionMatrixCellColor(matrixData.NormalizedMatrix[i, j])
                    });
                }
            }
            
            // Generate classification report
            await GenerateClassificationReport(matrixData);
        }
        
        ShowSuccessMessage("Confusion matrix generated successfully!");
    }
    catch (Exception ex)
    {
        ShowErrorMessage($"Confusion matrix generation failed: {ex.Message}");
    }
}

public class ConfusionMatrixData
{
    public string ActualClass { get; set; }
    public string PredictedClass { get; set; }
    public int Count { get; set; }
    public double Percentage { get; set; }
    public bool IsCorrectPrediction { get; set; }
    public string CellColor { get; set; }
    public string ToolTip => $"Actual: {ActualClass}\nPredicted: {PredictedClass}\nCount: {Count} ({Percentage:P1})";
}

// Generate ROC curve with AUC analysis
private async Task GenerateROCCurveAsync()
{
    try
    {
        CurrentChartType = "ROC Curve Analysis";
        
        if (MLManager is PythonMLManager manager && EvaluationResults != null)
        {
            // Generate ROC curve data for each class
            var rocData = await manager.Visualization.GenerateROCCurve(
                modelId: EvaluationResults.ModelId,
                multiClass: true,
                includeMicroAverage: true,
                includeMacroAverage: true);
            
            ROCCurveData.Clear();
            
            // Add data points for each class
            foreach (var classData in rocData.ClassCurves)
            {
                foreach (var point in classData.CurvePoints)
                {
                    ROCCurveData.Add(new ROCCurvePoint
                    {
                        ClassName = classData.ClassName,
                        FalsePositiveRate = point.FPR,
                        TruePositiveRate = point.TPR,
                        Threshold = point.Threshold,
                        AUC = classData.AUC,
                        CurveColor = GetROCCurveColor(classData.ClassName)
                    });
                }
            }
            
            // Update AUC summary statistics
            await UpdateAUCSummaryStatistics(rocData);
        }
        
        ShowSuccessMessage("ROC curve analysis completed!");
    }
    catch (Exception ex)
    {
        ShowErrorMessage($"ROC curve generation failed: {ex.Message}");
    }
}

public class ROCCurvePoint
{
    public string ClassName { get; set; }
    public double FalsePositiveRate { get; set; }
    public double TruePositiveRate { get; set; }
    public double Threshold { get; set; }
    public double AUC { get; set; }
    public string CurveColor { get; set; }
    public string ToolTip => $"{ClassName}\nTPR: {TruePositiveRate:F3}\nFPR: {FalsePositiveRate:F3}\nAUC: {AUC:F3}";
}

// Generate feature importance visualization
private async Task GenerateFeatureImportanceAsync()
{
    try
    {
        CurrentChartType = "Feature Importance Analysis";
        
        if (MLManager is PythonMLManager manager && EvaluationResults != null)
        {
            // Generate feature importance data
            var importanceData = await manager.Visualization.GenerateFeatureImportance(
                modelId: EvaluationResults.ModelId,
                method: "built_in", // or "permutation", "shap"
                topN: 20,
                includeNegative: true);
            
            FeatureImportanceData.Clear();
            
            // Sort by absolute importance and take top features
            var sortedFeatures = importanceData.Features
                .OrderByDescending(f => Math.Abs(f.Importance))
                .Take(20);
            
            foreach (var feature in sortedFeatures)
            {
                FeatureImportanceData.Add(new FeatureImportanceData
                {
                    FeatureName = feature.FeatureName,
                    Importance = feature.Importance,
                    AbsoluteImportance = Math.Abs(feature.Importance),
                    Rank = feature.Rank,
                    IsPositiveImportance = feature.Importance > 0,
                    BarColor = feature.Importance > 0 ? "#28a745" : "#dc3545", // Green for positive, red for negative
                    RelativeImportance = feature.Importance / importanceData.MaxImportance,
                    Description = GetFeatureDescription(feature.FeatureName)
                });
            }
            
            // Generate SHAP values if available
            await GenerateSHAPAnalysis(importanceData);
        }
        
        ShowSuccessMessage("Feature importance analysis completed!");
    }
    catch (Exception ex)
    {
        ShowErrorMessage($"Feature importance generation failed: {ex.Message}");
    }
}

public class FeatureImportanceData
{
    public string FeatureName { get; set; }
    public double Importance { get; set; }
    public double AbsoluteImportance { get; set; }
    public int Rank { get; set; }
    public bool IsPositiveImportance { get; set; }
    public string BarColor { get; set; }
    public double RelativeImportance { get; set; }
    public string Description { get; set; }
    public string ToolTip => $"{FeatureName}\nImportance: {Importance:F4}\nRank: {Rank}";
}

// Generate SHAP (SHapley Additive exPlanations) analysis
private async Task GenerateSHAPAnalysis(FeatureImportanceResult importanceData)
{
    try
    {
        if (MLManager is PythonMLManager manager)
        {
            // Generate SHAP values for model interpretability
            var shapData = await manager.Visualization.GenerateSHAPAnalysis(
                modelId: EvaluationResults.ModelId,
                analysisType: "summary", // or "waterfall", "force", "dependence"
                maxDisplay: 15,
                outputPath: GetChartOutputPath("shap_analysis.png"));
            
            // Update UI with SHAP insights
            if (shapData.Success)
            {
                CurrentChartType = "SHAP Analysis Completed";
                // Add SHAP-specific data to collections
                await UpdateSHAPVisualization(shapData);
            }
        }
    }
    catch (Exception ex)
    {
        Editor?.AddLogMessage("ModelEvaluationGraphsViewModel", 
            $"SHAP analysis failed: {ex.Message}", DateTime.Now, -1, null, Errors.Warning);
    }
}

Real-time Model Performance Tracking

// Real-time performance monitoring during training
public void StartRealTimeMonitoring(string modelId)
{
    try
    {
        // Setup timer for periodic updates
        _monitoringTimer = new Timer(UpdateChartsCallback, null, TimeSpan.Zero, TimeSpan.FromSeconds(5));
        _currentModelId = modelId;
        IsRealTimeMonitoringActive = true;
        
        ShowInformationMessage("Real-time monitoring started");
    }
    catch (Exception ex)
    {
        ShowErrorMessage($"Failed to start monitoring: {ex.Message}");
    }
}

private async void UpdateChartsCallback(object state)
{
    try
    {
        if (!IsRealTimeMonitoringActive || string.IsNullOrEmpty(_currentModelId))
            return;
        
        await Dispatcher.BeginInvoke(async () =>
        {
            // Get latest training metrics
            var latestMetrics = await GetLatestTrainingMetrics(_currentModelId);
            
            if (latestMetrics != null)
            {
                // Update performance data collection
                UpdatePerformanceDataInRealTime(latestMetrics);
                
                // Update metric trends
                UpdateMetricTrends(latestMetrics);
                
                // Trigger chart refresh
                OnPropertyChanged(nameof(PerformanceData));
                OnPropertyChanged(nameof(MetricTrends));
            }
        });
    }
    catch (Exception ex)
    {
        Editor?.AddLogMessage("ModelEvaluationGraphsViewModel", 
            $"Real-time update failed: {ex.Message}", DateTime.Now, -1, null, Errors.Warning);
    }
}

private void UpdatePerformanceDataInRealTime(TrainingMetrics metrics)
{
    // Add new data points to the performance collection
    var timestamp = DateTime.Now;
    
    PerformanceData.Add(new ChartDataPoint
    {
        Label = "Training Accuracy",
        Value = metrics.TrainingAccuracy,
        Category = "Training",
        Timestamp = timestamp,
        Color = "#007bff"
    });
    
    PerformanceData.Add(new ChartDataPoint
    {
        Label = "Validation Accuracy", 
        Value = metrics.ValidationAccuracy,
        Category = "Validation",
        Timestamp = timestamp,
        Color = "#28a745"
    });
    
    PerformanceData.Add(new ChartDataPoint
    {
        Label = "Training Loss",
        Value = metrics.TrainingLoss,
        Category = "Loss",
        Timestamp = timestamp,
        Color = "#dc3545"
    });
    
    // Keep only last N data points to prevent memory issues
    const int maxDataPoints = 100;
    while (PerformanceData.Count > maxDataPoints)
    {
        PerformanceData.RemoveAt(0);
    }
}

// Export charts in multiple formats
private async Task ExportChartsAsync()
{
    try
    {
        CurrentChartType = "Exporting Charts";
        
        var exportFormats = new[] { "png", "pdf", "svg", "html" };
        var exportTasks = new List<Task>();
        
        foreach (var format in exportFormats)
        {
            if (ShouldExportFormat(format))
            {
                exportTasks.Add(ExportChartInFormat(format));
            }
        }
        
        await Task.WhenAll(exportTasks);
        
        // Generate comprehensive report
        await GenerateEvaluationReport();
        
        ShowSuccessMessage($"Charts exported successfully to {GetExportDirectory()}");
    }
    catch (Exception ex)
    {
        ShowErrorMessage($"Chart export failed: {ex.Message}");
    }
}

private async Task ExportChartInFormat(string format)
{
    if (MLManager is PythonMLManager manager)
    {
        var exportConfig = new ChartExportConfiguration
        {
            Format = format,
            DPI = 300,
            Width = 12,
            Height = 8,
            IncludeTitle = true,
            IncludeLegend = true,
            Theme = CurrentTheme.ToString().ToLower(),
            OutputDirectory = GetExportDirectory()
        };
        
        // Export all chart types
        await manager.Visualization.ExportPerformanceCharts(EvaluationResults.ModelId, exportConfig);
        await manager.Visualization.ExportConfusionMatrix(EvaluationResults.ModelId, exportConfig);
        await manager.Visualization.ExportROCCurve(EvaluationResults.ModelId, exportConfig);
        await manager.Visualization.ExportFeatureImportance(EvaluationResults.ModelId, exportConfig);
    }
}

// Generate comprehensive evaluation report
private async Task GenerateEvaluationReport()
{
    var reportBuilder = new StringBuilder();
    
    // Header
    reportBuilder.AppendLine("# Model Evaluation Report");
    reportBuilder.AppendLine($"**Generated:** {DateTime.Now:yyyy-MM-dd HH:mm:ss}");
    reportBuilder.AppendLine($"**Model ID:** {EvaluationResults?.ModelId}");
    reportBuilder.AppendLine($"**Algorithm:** {EvaluationResults?.Algorithm}");
    reportBuilder.AppendLine();
    
    // Performance Summary
    reportBuilder.AppendLine("## Performance Summary");
    if (EvaluationResults != null)
    {
        reportBuilder.AppendLine($"- **Accuracy:** {EvaluationResults.Accuracy:P2}");
        reportBuilder.AppendLine($"- **Precision:** {EvaluationResults.Precision:F4}");
        reportBuilder.AppendLine($"- **Recall:** {EvaluationResults.Recall:F4}");
        reportBuilder.AppendLine($"- **F1-Score:** {EvaluationResults.F1Score:F4}");
        reportBuilder.AppendLine($"- **AUC:** {EvaluationResults.AUC:F4}");
    }
    reportBuilder.AppendLine();
    
    // Feature Importance Summary
    reportBuilder.AppendLine("## Top Features");
    foreach (var feature in FeatureImportanceData.Take(10))
    {
        reportBuilder.AppendLine($"- **{feature.FeatureName}:** {feature.Importance:F4}");
    }
    
    // Save report
    var reportPath = Path.Combine(GetExportDirectory(), "evaluation_report.md");
    await File.WriteAllTextAsync(reportPath, reportBuilder.ToString());
}
