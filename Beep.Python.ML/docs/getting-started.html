‚öôÔ∏è Basic Setup

Initialize ML Manager

using Beep.Python.ML;
using Beep.Python.Model;
using TheTechIdea.Beep.Container.Services;

// Initialize core services
var beepService = new BeepService();
var pythonRuntime = new PythonNetRunTimeManager(beepService);
var sessionInfo = new PythonSessionInfo 
{ 
    Username = "ml_user",
    SessionId = Guid.NewGuid().ToString(),
    CreateTime = DateTime.UtcNow
};

// Create ML manager
var mlManager = new PythonMLManager(beepService, pythonRuntime, sessionInfo);

// Configure session for user
bool sessionConfigured = mlManager.ConfigureMLSessionForUser("ml_user", environmentId: null);

if (sessionConfigured)
{
    Console.WriteLine("‚úÖ ML Manager initialized successfully!");
}
else
{
    Console.WriteLine("‚ùå Failed to configure ML session");
}
Session Configuration Required

Always configure sessions before performing ML operations. This ensures proper Python environment isolation and resource management.

üöÄ Your First ML Pipeline

Complete ML Workflow Example

public async Task RunFirstMLPipeline()
{
    // 1. Initialize ML Manager
    var mlManager = new PythonMLManager(beepService, pythonRuntime, sessionInfo);
    mlManager.ConfigureMLSessionForUser("data_scientist");

    try
    {
        // 2. Load and validate data
        Console.WriteLine("üìä Loading data...");
        var features = mlManager.ValidateAndPreviewData("data/iris.csv", numRows: 5);
        Console.WriteLine($"Found {features.Length} features: {string.Join(", ", features)}");

        // Load the full dataset
        var loadedFeatures = mlManager.LoadData("data/iris.csv");
        Console.WriteLine($"‚úÖ Loaded dataset with {loadedFeatures.Length} features");

        // 3. Use assistant classes for data preprocessing
        if (mlManager is PythonMLManager manager)
        {
            Console.WriteLine("üßπ Preprocessing data...");
            
            // Data preprocessing
            manager.DataPreprocessing.ImputeMissingValues("mean");
            manager.DataPreprocessing.StandardizeData();
            
            // Handle categorical data (if any)
            var categoricalFeatures = new[] { "species" };
            manager.CategoricalEncoding.LabelEncode(categoricalFeatures);
            
            // Feature engineering
            manager.FeatureEngineering.GeneratePolynomialFeatures(degree: 2, interactionOnly: true);
            
            // Data quality checks
            manager.DataCleaning.RemoveOutliers(zThreshold: 3.0);
            manager.DataCleaning.DropDuplicates();
        }

        // 4. Train a classification model
        Console.WriteLine("ü§ñ Training model...");
        var modelId = Guid.NewGuid().ToString();
        var parameters = new Dictionary<string, object>
        {
            ["n_estimators"] = 100,
            ["max_depth"] = 10,
            ["random_state"] = 42,
            ["class_weight"] = "balanced"
        };

        mlManager.TrainModel(
            modelId, 
            MachineLearningAlgorithm.RandomForestClassifier, 
            parameters, 
            loadedFeatures.Take(4).ToArray(), // Use first 4 features
            "species"); // Target column

        Console.WriteLine("‚úÖ Model training completed!");

        // 5. Evaluate the model
        Console.WriteLine("üìà Evaluating model...");
        var scores = mlManager.GetModelClassificationScore(modelId);
        Console.WriteLine($"Model Performance:");
        Console.WriteLine($"  Accuracy: {scores.Item1:F4}");
        Console.WriteLine($"  F1-Score: {scores.Item2:F4}");

        // 6. Make predictions
        Console.WriteLine("üîÆ Making predictions...");
        var predictions = mlManager.PredictClassification(loadedFeatures.Take(4).ToArray());
        Console.WriteLine($"Generated {predictions.ToString().Split(',').Length} predictions");

        // 7. Advanced operations with assistants
        if (mlManager is PythonMLManager advancedManager)
        {
            Console.WriteLine("üìä Creating visualizations...");
            
            // Cross-validation
            advancedManager.CrossValidation.PerformCrossValidation(modelId, numFolds: 5);
            
            // Create visualizations
            advancedManager.Visualization.CreateROC(modelId, "outputs/roc_curve.png");
            advancedManager.Visualization.CreateConfusionMatrix(modelId, "outputs/confusion_matrix.png");
            advancedManager.Visualization.CreateFeatureImportance(modelId, "outputs/feature_importance.png");
            
            // Generate comprehensive report
            advancedManager.Visualization.GenerateEvaluationReport(modelId, "outputs/ml_report.html");
        }

        Console.WriteLine("üéâ ML Pipeline completed successfully!");
        Console.WriteLine("üìÅ Check the 'outputs' folder for generated visualizations and reports");

    }
    catch (Exception ex)
    {
        Console.WriteLine($"‚ùå Pipeline failed: {ex.Message}");
        Console.WriteLine($"Stack trace: {ex.StackTrace}");
    }
}

// Cast to access assistant classes

{
    // Handle missing values
    manager.DataPreprocessing.ImputeMissingValues("median");

3. Advanced Feature Engineering Pattern

if (mlManager is PythonMLManager manager)
{
    // Create polynomial features
    manager.FeatureEngineering.GeneratePolynomialFeatures(
        selectedFeatures: new[] { "age", "income", "score" },
        degree: 2,
        interactionOnly: true);

}

‚ú® Best Practices

Development Best Practices
  • Session Management: Always configure sessions before ML operations
  • Error Handling: Wrap ML operations in try-catch blocks
  • Assistant Access: Cast to PythonMLManager to access assistants
  • Resource Cleanup: Properly dispose of managers when done
  • Async Operations: Use async methods for long-running tasks
Performance Tips
  • Data Size: Consider data size and memory usage
  • Model Complexity: Start simple, then increase complexity
  • Validation: Always validate your models properly
  • Feature Engineering: Invest time in good features
  • Monitoring: Monitor model performance in production
