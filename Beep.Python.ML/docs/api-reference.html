Usage Example

Basic Interface Usage

// Initialize ML manager
IPythonMLManager mlManager = new PythonMLManager(beepService, pythonRuntime, sessionInfo);

// Configure session
mlManager.ConfigureMLSessionForUser("data_scientist");

// Load and validate data
var features = mlManager.ValidateAndPreviewData("data.csv", 10);
var loadedFeatures = mlManager.LoadData("data.csv", features.Take(5).ToArray());

// Train model
var parameters = new Dictionary<string, object>
{
    ["n_estimators"] = 100,
    ["max_depth"] = 10,
    ["random_state"] = 42
};

mlManager.TrainModel("model1", MachineLearningAlgorithm.RandomForestClassifier, 
                    parameters, loadedFeatures, "target");

// Evaluate model
var scores = mlManager.GetModelClassificationScore("model1");
Console.WriteLine($"Accuracy: {scores.Item1:F4}, F1-Score: {scores.Item2:F4}");

// Make predictions
var predictions = mlManager.PredictClassification(loadedFeatures);

        {
            ["n_estimators"] = 100,
            ["max_depth"] = 10,

        };
        break;
}

Error Handling Best Practices

try
{
    // Configure session first
    if (!mlManager.ConfigureMLSessionForUser("user1"))
    {
        throw new InvalidOperationException("Failed to configure ML session");
    }

    // Validate data file exists
    if (!File.Exists(dataFilePath))
    {
        throw new FileNotFoundException($"Data file not found: {dataFilePath}");
    }

    // Load and validate data
    var features = mlManager.ValidateAndPreviewData(dataFilePath);
    if (features == null || features.Length == 0)
    {
        throw new InvalidOperationException("No valid features found in data file");
    }

    // Use assistant classes with error handling
    if (mlManager is PythonMLManager manager)
    {
        manager.DataPreprocessing.ImputeMissingValues("mean");
        manager.DataPreprocessing.StandardizeData();
        manager.FeatureEngineering.GeneratePolynomialFeatures(degree: 2);
    }

    // Train model with parameter validation
    var parameters = new Dictionary<string, object>
    {
        ["n_estimators"] = 100,
        ["max_depth"] = 10,
        ["random_state"] = 42
    };

    mlManager.TrainModel("model1", MachineLearningAlgorithm.RandomForestClassifier, 
                        parameters, features, "target");

    Console.WriteLine("‚úÖ ML pipeline completed successfully");
}
catch (InvalidOperationException ex)
{
    Console.WriteLine($"‚ùå Configuration error: {ex.Message}");
}
catch (FileNotFoundException ex)
{
    Console.WriteLine($"‚ùå File error: {ex.Message}");
}
catch (ArgumentException ex)
{
    Console.WriteLine($"‚ùå Parameter error: {ex.Message}");
}
catch (Exception ex)
{
    Console.WriteLine($"‚ùå Unexpected error: {ex.Message}");
    Console.WriteLine($"Stack trace: {ex.StackTrace}");
}

‚ö° Performance Considerations

Performance Tips
  • Session Reuse: Configure sessions once and reuse for multiple operations
  • Lazy Initialization: Assistant classes are created only when accessed
  • Script Caching: Python scripts are cached for optimal performance
  • Memory Management: Dispose of managers properly after use
  • Async Operations: Use async methods for long-running ML operations
  • Data Size: Consider data size when choosing algorithms and parameters
Memory Usage
  • Large Datasets: Monitor memory usage with large datasets
  • Model Complexity: Complex models require more memory
  • Feature Engineering: Polynomial features can exponentially increase memory usage
  • Batch Processing: Consider batch processing for very large datasets
