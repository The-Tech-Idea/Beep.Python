Overview

Virtual environments are isolated Python environments that allow you to manage dependencies, packages, and Python versions independently for different projects or users. Beep.Python.Runtime.PythonNet provides comprehensive virtual environment support through the PythonVirtualEnvManager class.

?? Isolation

Complete package and dependency isolation between environments

?? Dual Support

Works with both standard Python venv and Conda environments

?? Multi-User

User-specific environments with session management

?? Cross-Platform

Windows, Linux, and macOS compatibility

Environment Types

The library supports multiple types of Python environments, each with specific characteristics:

?? Standard Python Virtual Environments

Created using Python's built-in venv module. Ideal for standard Python development.

  • Package Manager: pip
  • Isolation Level: Package-level
  • Best For: General Python development, web applications

??? Conda Environments

Created using Anaconda/Miniconda. Excellent for data science and scientific computing.

  • Package Manager: conda (with pip fallback)
  • Isolation Level: System-level with binary dependencies
  • Best For: Data science, machine learning, scientific computing

??? System Python

Uses the system-wide Python installation. Not recommended for production.

  • Package Manager: pip (system-wide)
  • Isolation Level: None
  • Best For: Testing, development, single-user scenarios

Creating Environments

Basic Environment Creation

Creating a Standard Python Environment

// Get the virtual environment manager
var envManager = runtimeManager.VirtualEnvmanager;

// Define the new environment
var newEnvironment = new PythonVirtualEnvironment
{
    Name = "my-project-env",
    Path = @"C:\PythonEnvironments\my-project-env",
    PythonConfigID = pythonRuntime.ID,
    BaseInterpreterPath = pythonRuntime.RuntimePath,
    CreatedBy = Environment.UserName,
    PythonBinary = PythonBinary.Pip,
    EnvironmentType = PythonEnvironmentType.VirtualEnv
};

// Create the environment
bool created = envManager.CreateVirtualEnvironment(pythonRuntime, newEnvironment);

if (created)
{
    Console.WriteLine($"Environment '{newEnvironment.Name}' created successfully!");
    envManager.AddToManagedEnvironments(newEnvironment);
}
else
{
    Console.WriteLine("Failed to create environment.");
}

Working with Managed Environments

// Find environments by user

var userEnvironments = allEnvironments
    .Where(e => e.CreatedBy == targetUser)

Environment Activation and Initialization

Activating and Using Environments

// Initialize Python with a specific environment
bool initialized = runtimeManager.Initialize(
    pythonRuntime, 
    foundEnv.Path, 
    foundEnv.Name, 
    PythonEngineMode.SingleUser);

if (initialized)
{
    Console.WriteLine($"Python initialized with environment: {foundEnv.Name}");
    
    // Initialize the environment's Python paths and variables
    envManager.InitializePythonEnvironment(foundEnv);
    
    // Get a session associated with this environment
    var session = runtimeManager.GetOrCreateDefaultSession();
    
    // Execute code in the environment context
    var executeManager = runtimeManager.ExecuteManager;
    var result = await executeManager.ExecuteCodeAsync(
        "import sys; print('Python path:', sys.executable)", 
        session);
    
    Console.WriteLine(result.Output);
}

Environment Switching

Switching Between Environments

// Switch to a different environment
var targetEnvironment = envManager.ManagedVirtualEnvironments
    .FirstOrDefault(e => e.Name == "data-science-env");

if (targetEnvironment != null)
{
    // Restart Python with the new environment
    bool restarted = runtimeManager.RestartWithEnvironment(targetEnvironment);
    
    if (restarted)
    {
        Console.WriteLine($"Switched to environment: {targetEnvironment.Name}");
        
        // Verify the switch
        var session = runtimeManager.GetOrCreateDefaultSession();
        var result = await executeManager.ExecuteCodeAsync(
            @"
import sys
import os
print(f'Active environment: {os.environ.get(""VIRTUAL_ENV"", ""System"")}')
print(f'Python executable: {sys.executable}')
", 
            session);
        
        Console.WriteLine(result.Output);
    }
}

Environment Lifecycle

Environment Cleanup and Maintenance

Automatic Environment Cleanup

// Perform cleanup of unused environments
var cleanupCriteria = new EnvironmentCleanupCriteria
{
    MaxAge = TimeSpan.FromDays(30),
    RequireEmptySessions = true,
    OnlyTemporaryEnvironments = false
};

envManager.PerformEnvironmentCleanup(cleanupCriteria);

// Manual cleanup of specific environment
var environmentToCleanup = envManager.ManagedVirtualEnvironments
    .FirstOrDefault(e => e.Name == "temp-environment");

if (environmentToCleanup != null)
{
    // Ensure no active sessions
    if (environmentToCleanup.Sessions.Count == 0)
    {
        bool removed = envManager.RemoveVirtualEnvironment(environmentToCleanup.ID);
        if (removed)
        {
            Console.WriteLine($"Environment '{environmentToCleanup.Name}' removed successfully");
        }
    }
}

Cross-Platform Support

Platform-Specific Considerations

?? Windows
  • Python executables: python.exe, Scripts/ directory
  • Conda environments: Often in %USERPROFILE%\Anaconda3\envs\
  • Virtual environments: venv\Scripts\activate.bat
?? Linux/macOS
  • Python executables: python3, bin/ directory
  • Conda environments: Often in ~/anaconda3/envs/
  • Virtual environments: venv/bin/activate

Platform-Aware Environment Creation

// Cross-platform environment creation
string GetPlatformSpecificPath(string basePath, string envName)
{
    if (Environment.OSVersion.Platform == PlatformID.Win32NT)
    {
        return Path.Combine(basePath, envName);
    }
    else
    {
        // Linux/macOS - often use hidden directories
        string homeDir = Environment.GetFolderPath(Environment.SpecialFolder.UserProfile);
        return Path.Combine(homeDir, ".python-environments", envName);
    }
}

// Create environment with platform-appropriate paths
var crossPlatformEnv = new PythonVirtualEnvironment
{
    Name = "cross-platform-env",
    Path = GetPlatformSpecificPath("/opt/python-envs", "cross-platform-env"),
    PythonConfigID = pythonRuntime.ID,
    BaseInterpreterPath = pythonRuntime.RuntimePath,
    CreatedBy = Environment.UserName,
    PythonBinary = PythonBinary.Pip,
    EnvironmentType = PythonEnvironmentType.VirtualEnv
};

bool created = envManager.CreateVirtualEnvironment(pythonRuntime, crossPlatformEnv);

Environment Best Practices Example

// Best practice environment setup
public class EnvironmentBestPractices
{
    private readonly PythonVirtualEnvManager _envManager;
    
    public async Task CreateProjectEnvironment(
        string projectName, 
        string username, 
        List requiredPackages)
    {
        // 1. Use descriptive naming convention
        string envName = $"{projectName}-{username}-{DateTime.Now:yyyyMM}";
        
        // 2. Create isolated user directory
        string userEnvDir = Path.Combine(
            GetUserEnvironmentsPath(), 
            username);
        
        if (!Directory.Exists(userEnvDir))
        {
            Directory.CreateDirectory(userEnvDir);
        }
        
        // 3. Generate requirements file
        string requirementsPath = Path.Combine(userEnvDir, $"{envName}-requirements.txt");
        await File.WriteAllLinesAsync(requirementsPath, requiredPackages);
        
        // 4. Create environment with metadata
        var environment = new PythonVirtualEnvironment
        {
            Name = envName,
            Path = Path.Combine(userEnvDir, envName),
            CreatedBy = username,
            CreatedOn = DateTime.Now,
            RequirementsFile = requirementsPath,
            Metadata = new Dictionary
            {
                ["ProjectName"] = projectName,
                ["AutoCleanup"] = true,
                ["MaxAge"] = TimeSpan.FromDays(90)
            }
        };
        
        return environment;
    }
    
    private string GetUserEnvironmentsPath()
    {
        // Platform-specific user environment paths
        if (Environment.OSVersion.Platform == PlatformID.Win32NT)
        {
            return Path.Combine(
                Environment.GetFolderPath(Environment.SpecialFolder.LocalApplicationData),
                "PythonEnvironments");
        }
        else
        {
            return Path.Combine(
                Environment.GetFolderPath(Environment.SpecialFolder.UserProfile),
                ".python-environments");
        }
    }
}

Troubleshooting

Common Issues and Solutions

? Issue: Environment Creation Fails
  • Check Permissions: Ensure write access to target directory
  • Verify Python Path: Confirm base Python installation is valid
  • Check Disk Space: Ensure sufficient disk space available
  • Path Length: On Windows, avoid paths longer than 260 characters
? Issue: Package Installation Fails
  • Network Connectivity: Check internet connection and proxy settings
  • Package Index: Verify PyPI or conda repositories are accessible
  • Dependencies: Resolve conflicting package dependencies
  • Compiler Requirements: Install necessary build tools for native packages
? Issue: Environment Activation Problems
  • Path Configuration: Verify environment paths are correctly set
  • Python Executable: Check if Python executable exists and is functional
  • Environment Variables: Ensure VIRTUAL_ENV and PATH are set correctly
  • Session State: Restart Python runtime if environment switching fails

Diagnostic and Repair Utilities

// Comprehensive environment diagnostics
public class EnvironmentDiagnostics
{
    public static void DiagnoseEnvironment(PythonVirtualEnvironment env)
    {
        Console.WriteLine($"Diagnosing environment: {env.Name}");
        
        // Check basic structure
        CheckEnvironmentStructure(env);
        
        // Check Python executable
        CheckPythonExecutable(env);
        
        // Check package manager
        CheckPackageManager(env);
        
        // Check environment variables
        CheckEnvironmentVariables(env);
    }
    
    private static void CheckEnvironmentStructure(PythonVirtualEnvironment env)
    {
        bool pathExists = Directory.Exists(env.Path);
        Console.WriteLine($"  Environment path exists: {pathExists}");
        
        if (pathExists)
        {
            var subdirs = Directory.GetDirectories(env.Path);
            Console.WriteLine($"  Subdirectories: {string.Join(", ", subdirs.Select(Path.GetFileName))}");
        }
    }
    
    private static void CheckPythonExecutable(PythonVirtualEnvironment env)
    {
        string pythonExe = env.PythonBinary == PythonBinary.Conda
            ? Path.Combine(env.Path, "python.exe")
            : Path.Combine(env.Path, "Scripts", "python.exe");
            
        bool exeExists = File.Exists(pythonExe);
        Console.WriteLine($"  Python executable exists: {exeExists}");
        
        if (exeExists)
        {
            try
            {
                var process = Process.Start(new ProcessStartInfo
                {
                    FileName = pythonExe,
                    Arguments = "--version",
                    UseShellExecute = false,
                    RedirectStandardOutput = true
                });
                
                string version = process.StandardOutput.ReadToEnd().Trim();
                Console.WriteLine($"  Python version: {version}");
            }
            catch (Exception ex)
            {
                Console.WriteLine($"  Error checking Python version: {ex.Message}");
            }
        }
    }
    
    private static void CheckPackageManager(PythonVirtualEnvironment env)
    {
        string packageManager = env.PythonBinary == PythonBinary.Conda ? "conda" : "pip";
        Console.WriteLine($"  Package manager: {packageManager}");
        
        // Additional package manager checks would go here
    }
    
    private static void CheckEnvironmentVariables(PythonVirtualEnvironment env)
    {
        var expectedVars = new Dictionary
        {
            ["VIRTUAL_ENV"] = env.Path,
            ["PYTHONHOME"] = env.Path
        };
        
        foreach (var kvp in expectedVars)
        {
            string actualValue = Environment.GetEnvironmentVariable(kvp.Key);
            bool matches = actualValue == kvp.Value;
            Console.WriteLine($"  {kvp.Key}: {(matches ? "?" : "?")} (Expected: {kvp.Value}, Actual: {actualValue})");
        }
    }
}
