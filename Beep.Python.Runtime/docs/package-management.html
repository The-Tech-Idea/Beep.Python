Overview

The PythonPackageManager provides comprehensive package management capabilities for Python environments. It supports both pip and conda package managers, automatic dependency resolution, categorized package organization, and bulk operations for efficient development workflows.

?? Dual Manager Support

Seamless integration with both pip and conda package managers

Efficient bulk installation and management of package sets

?? Progress Tracking

Real-time progress reporting for all package operations

Package Managers

?? Pip Package Manager

Standard Python package installer with access to PyPI (Python Package Index).

  • Repository: PyPI (pypi.org)
  • Package Format: Wheels (.whl) and source distributions
  • Dependency Resolution: Basic with conflict detection
  • Best For: General Python packages, web frameworks, utilities

??? Conda Package Manager

Package manager focused on data science with binary dependency management.

  • Repository: Conda-forge, Anaconda, custom channels
  • Package Format: Conda packages with binary dependencies
  • Dependency Resolution: Advanced with SAT solver
  • Best For: Data science, scientific computing, native libraries

Initializing Package Manager

// Initialize package manager for specific environment and session
var packageManager = new PythonPackageManager(
    beepService,           // Beep service for logging and configuration
    runtimeManager,        // Python runtime manager
    environmentManager,    // Virtual environment manager
    environment,          // Target virtual environment
    session               // Python session for operations
);

// Set up progress reporting
packageManager.Progress = new Progress(args => 
{
    Console.WriteLine($"Package Operation: {args.Messege}");
});

Console.WriteLine($"Package manager initialized for environment: {environment.Name}");
Console.WriteLine($"Package manager type: {environment.PythonBinary}");

Installation Methods

Single Package Installation

Installing Individual Packages

// Install a specific package
bool success = await packageManager.InstallNewPackageAsync("numpy");

if (success)
{
    Console.WriteLine("NumPy installed successfully!");
}

// Install with specific version
success = await packageManager.InstallNewPackageAsync("pandas==1.5.0");

// Install with version constraints
success = await packageManager.InstallNewPackageAsync("matplotlib>=3.0.0");

// Install from specific index
success = await packageManager.InstallNewPackageAsync(
    "tensorflow", 
    extraIndexUrl: "https://pypi.org/simple/");

Bulk Package Installation

Installing Multiple Packages

// Define a list of packages to install
var packagesToInstall = new List
{
    "numpy",
    "pandas",
    "matplotlib",
    "scikit-learn",
    "jupyter",
    "seaborn"
};

// Install all packages
var results = await packageManager.InstallPackagesAsync(packagesToInstall);

// Check results
foreach (var result in results)
{
    string status = result.Value ? "? Success" : "? Failed";
    Console.WriteLine($"{result.Key}: {status}");
}

// Count successful installations
int successCount = results.Count(r => r.Value);
Console.WriteLine($"Successfully installed {successCount} out of {packagesToInstall.Count} packages");

Package Upgrading and Management

Upgrading and Managing Packages

// Upgrade a specific package to the latest version
bool upgraded = await packageManager.UpgradePackageAsync("numpy");

// Upgrade all packages (use with caution)
var allPackages = await packageManager.GetInstalledPackagesAsync();
var upgradeResults = await packageManager.UpgradePackagesAsync(
    allPackages.Select(p => p.PackageName).ToList());

// Get package information
var packageInfo = await packageManager.GetPackageInfoAsync("pandas");

{
    Console.WriteLine($"Package: {packageInfo.PackageName}");

    Console.WriteLine($"Category: {packageInfo.Category}");
}

Requirements Management

Requirements File Generation

Creating and Managing Requirements Files

// Generate requirements.txt for current environment
string requirementsPath = Path.Combine(environment.Path, "requirements.txt");
bool generated = await packageManager.GenerateRequirementsFile(requirementsPath);

if (generated)
{
    Console.WriteLine($"Requirements file generated: {requirementsPath}");
    
    // Read and display the requirements
    string[] requirements = await File.ReadAllLinesAsync(requirementsPath);
    Console.WriteLine("Generated requirements:");
    foreach (string req in requirements)
    {
        Console.WriteLine($"  {req}");
    }
}

// Generate requirements with additional options
var requirementsOptions = new RequirementsGenerationOptions
{
    IncludeVersions = true,
    IncludeHashes = false,
    ExcludeDevelopmentPackages = true,
    SortAlphabetically = true
};

bool customGenerated = await packageManager.GenerateRequirementsFile(
    requirementsPath, 
    requirementsOptions);

Installing from Requirements

Installing from Requirements Files

// Install from requirements.txt
string requirementsFile = @"C:\Projects\MyProject\requirements.txt";
bool installed = await packageManager.InstallFromRequirementsAsync(requirementsFile);

if (installed)
{
    Console.WriteLine("All packages from requirements.txt installed successfully!");
}

// Install from multiple requirements files
var requirementsFiles = new List
{
    @"C:\Projects\MyProject\requirements.txt",
    @"C:\Projects\MyProject\requirements-dev.txt",
    @"C:\Projects\MyProject\requirements-test.txt"
};

foreach (string reqFile in requirementsFiles)
{
    if (File.Exists(reqFile))
    {
        Console.WriteLine($"Installing from {Path.GetFileName(reqFile)}...");
        await packageManager.InstallFromRequirementsAsync(reqFile);
    }
}

// Install with upgrade strategy
await packageManager.InstallFromRequirementsAsync(
    requirementsFile, 
    upgradeStrategy: UpgradeStrategy.UpgradeIfNewer);

Predefined Package Sets

Available Package Sets

Working with Package Sets

// Get available package sets
var availableSets = packageManager.GetAvailablePackageSets();

        {
            Console.WriteLine($"  - {package.PackageName}");
        }
    }
}

Dependency Analysis

Analyzing Package Dependencies

// Analyze dependencies for a package before installation
var dependencyAnalysis = await packageManager.AnalyzeDependencies("scikit-learn");

Console.WriteLine($"Package: {dependencyAnalysis.PackageName}");
Console.WriteLine($"Direct dependencies: {dependencyAnalysis.DirectDependencies.Count}");
Console.WriteLine($"Total dependencies: {dependencyAnalysis.AllDependencies.Count}");

Console.WriteLine("Direct dependencies:");
foreach (var dep in dependencyAnalysis.DirectDependencies)
{
    Console.WriteLine($"  - {dep.PackageName} {dep.VersionConstraint}");
}

// Check for conflicts
var conflicts = await packageManager.CheckForConflicts(dependencyAnalysis.AllDependencies);

if (conflicts.Any())
{
    Console.WriteLine("Potential conflicts detected:");
    foreach (var conflict in conflicts)
    {
        Console.WriteLine($"  {conflict.PackageName}: {conflict.ConflictDescription}");
    }
}
else
{
    Console.WriteLine("No conflicts detected");
}

Dependency Resolution Strategies

Advanced Dependency Resolution

// Configure dependency resolution strategy
var resolutionStrategy = new DependencyResolutionStrategy
{
    PreferLatestVersions = true,
    AllowPreReleases = false,
    UpgradeStrategy = UpgradeStrategy.UpgradeIfNewer,
    ConflictResolution = ConflictResolution.PreferExisting
};

packageManager.SetDependencyResolutionStrategy(resolutionStrategy);

// Resolve dependencies with strategy
var packages = new List { "pandas", "matplotlib", "scikit-learn" };
var resolvedPackages = await packageManager.ResolveDependencies(packages);

Console.WriteLine("Resolved installation order:");
for (int i = 0; i < resolvedPackages.Count; i++)
{
    Console.WriteLine($"{i + 1}. {resolvedPackages[i].PackageName} {resolvedPackages[i].Version}");
}

// Dry run installation to preview changes
var dryRunResult = await packageManager.DryRunInstallation(packages);

Console.WriteLine("Dry run results:");
Console.WriteLine($"Packages to install: {dryRunResult.ToInstall.Count}");
Console.WriteLine($"Packages to upgrade: {dryRunResult.ToUpgrade.Count}");
Console.WriteLine($"Total download size: {dryRunResult.EstimatedDownloadSize}");
Console.WriteLine($"Estimated install time: {dryRunResult.EstimatedInstallTime}");

Automated Updates

Setting Up Automated Package Updates

// Configure automatic update policy
var updatePolicy = new PackageUpdatePolicy
{
    UpdateFrequency = UpdateFrequency.Weekly,
    UpdateLevel = UpdateLevel.SecurityAndBugFixes,
    ExcludePackages = new List { "tensorflow", "pytorch" }, // Pin critical packages
    RequireApproval = true,
    BackupBeforeUpdate = true
};

packageManager.SetUpdatePolicy(updatePolicy);

// Perform automatic updates
var updateResults = await packageManager.PerformAutomaticUpdates();

Console.WriteLine("Automatic Update Results:");
Console.WriteLine($"Updates available: {updateResults.UpdatesAvailable}");
Console.WriteLine($"Updates applied: {updateResults.UpdatesApplied}");
Console.WriteLine($"Updates skipped: {updateResults.UpdatesSkipped}");
Console.WriteLine($"Errors encountered: {updateResults.Errors.Count}");

// Schedule periodic updates
var updateScheduler = new PackageUpdateScheduler(packageManager);
updateScheduler.ScheduleUpdates(updatePolicy);

Console.WriteLine("Automatic updates scheduled");

Troubleshooting

Common Package Issues

? Issue: Package Installation Fails
  • Check Network: Verify internet connectivity and proxy settings
  • Update Package Index: Refresh pip/conda package index
  • Clear Cache: Clear pip or conda cache to resolve corruption
  • Check Dependencies: Manually resolve conflicting dependencies
? Issue: Dependency Conflicts
  • Version Pinning: Pin conflicting packages to compatible versions
  • Separate Environments: Use different environments for incompatible packages
  • Alternative Packages: Consider alternative packages with fewer conflicts
  • Manual Resolution: Manually specify package versions
? Issue: Slow Package Operations
  • Mirror Selection: Use faster package mirrors or CDNs
  • Parallel Downloads: Enable parallel downloads if supported
  • Binary Packages: Prefer binary packages over source compilation
  • Local Cache: Configure local package caching

Package Troubleshooting Tools

// Comprehensive package environment diagnostics
public class PackageDiagnostics
{
    public static async Task DiagnosePackageEnvironment(PythonPackageManager packageManager)
    {
        Console.WriteLine("Running package environment diagnostics...");
        
        // Check package manager health
        var healthCheck = await packageManager.PerformHealthCheck();
        Console.WriteLine($"Package manager health: {(healthCheck.IsHealthy ? "?" : "?")}");
        
        if (!healthCheck.IsHealthy)
        {
            Console.WriteLine("Issues found:");
            foreach (var issue in healthCheck.Issues)
            {
                Console.WriteLine($"  - {issue}");
            }
        }
        
        // Check repository connectivity
        var repoCheck = await packageManager.CheckRepositoryConnectivity();
        Console.WriteLine($"Repository connectivity: {(repoCheck.CanConnect ? "?" : "?")}");
        
        // Verify package integrity
        var integrityCheck = await packageManager.VerifyPackageIntegrity();
        Console.WriteLine($"Package integrity: {integrityCheck.HealthyPackages}/{integrityCheck.TotalPackages} packages OK");
        
        if (integrityCheck.CorruptedPackages.Any())
        {
            Console.WriteLine("Corrupted packages:");
            foreach (var package in integrityCheck.CorruptedPackages)
            {
                Console.WriteLine($"  - {package}");
            }
        }
        
        // Check for orphaned packages
        var orphanedPackages = await packageManager.FindOrphanedPackages();
        Console.WriteLine($"Orphaned packages: {orphanedPackages.Count}");
        
        // Performance metrics
        var perfMetrics = await packageManager.GetPerformanceMetrics();
        Console.WriteLine($"Average install time: {perfMetrics.AverageInstallTime}");
        Console.WriteLine($"Cache hit ratio: {perfMetrics.CacheHitRatio:P2}");
    }
    
    public static async Task RepairPackageEnvironment(PythonPackageManager packageManager)
    {
        Console.WriteLine("Attempting to repair package environment...");
        
        // Clear caches
        await packageManager.ClearAllCaches();
        Console.WriteLine("Caches cleared");
        
        // Update package index
        await packageManager.UpdatePackageIndex();
        Console.WriteLine("Package index updated");
        
        // Repair corrupted packages
        var repairResults = await packageManager.RepairCorruptedPackages();
        Console.WriteLine($"Repaired {repairResults.RepairedCount} packages");
        
        // Clean orphaned packages
        var cleanupResults = await packageManager.CleanupOrphanedPackages();
        Console.WriteLine($"Cleaned up {cleanupResults.RemovedCount} orphaned packages");
        
        Console.WriteLine("Package environment repair completed");
    }
}
