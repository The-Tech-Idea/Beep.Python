?? PythonEnvironmentDiagnostics

Static utility class that provides comprehensive Python environment discovery and analysis capabilities.

Namespace: Beep.Python.RuntimeEngine.Helpers

Type: Static Class

Purpose: Python installation discovery, environment analysis, and diagnostic reporting

Key Features

  • System-wide Python Discovery: Automatically scans for all Python installations
  • Conda Environment Detection: Finds and analyzes Conda environments
  • Virtual Environment Support: Detects venv and virtualenv installations
  • Cross-Platform Compatibility: Works on Windows, Linux, and macOS
  • Diagnostic Reporting: Generates comprehensive diagnostic reports

Environment Discovery Example

// Scan system for all Python installations
var installations = PythonEnvironmentDiagnostics.LookForPythonInstallations();

Console.WriteLine($"Found {installations.Count} Python installations:");
foreach (var installation in installations)
{
    Console.WriteLine($"  Path: {installation.PythonPath}");
    Console.WriteLine($"  Version: {installation.PythonVersion}");
    Console.WriteLine($"  Type: {(installation.IsConda ? "Conda" : "Standard Python")}");
    Console.WriteLine($"  Packages: {installation.InstalledPackages.Count}");
    
    if (installation.Errors.Any())
    {
        Console.WriteLine($"  ?? Issues: {string.Join(", ", installation.Errors)}");
    }
    Console.WriteLine();
}

// Convert diagnostic reports to runtime objects
var runtimes = PythonEnvironmentDiagnostics.GetPythonRuntimesInstallations();
Console.WriteLine($"Configured {runtimes.Count} Python runtimes");

Conda Environment Discovery

// Get base Conda runtime
var baseRuntime = runtimes.FirstOrDefault(r => r.PackageType == PackageType.conda);

if (baseRuntime != null)
{
    // Find all Conda environments
    var condaEnvironments = PythonEnvironmentDiagnostics.GetCondaEnvironmentsFromRuntime(baseRuntime);
    
    Console.WriteLine($"Found {condaEnvironments.Count} Conda environments:");
    foreach (var env in condaEnvironments)
    {
        Console.WriteLine($"  Environment: {env.RuntimePath}");
        Console.WriteLine($"  Python Version: {env.PythonVersion}");
        Console.WriteLine($"  Message: {env.Message}");
    }
    
    // Generate package lists for environments
    foreach (var env in condaEnvironments)
    {
        string packageListPath = Path.Combine(env.RuntimePath, "package_list.txt");
        PythonEnvironmentDiagnostics.GeneratePackageListFile(env, packageListPath);
        Console.WriteLine($"Package list saved to: {packageListPath}");
    }
}

    • ?? PythonRunTimeDiagnostics

    Static utility class focused on Python runtime configuration, validation, and environment setup.

    Namespace: Beep.Python.RuntimeEngine.Helpers

    Type: Static Class

    Purpose: Runtime configuration, validation, and environment management

    Key Features

    • Installation Validation: Verifies Python installations and dependencies
    • Architecture Detection: Identifies 32-bit vs 64-bit installations
    • Version Analysis: Extracts and validates Python versions
    • Package Type Detection: Identifies pip vs conda environments
    • Network Connectivity: Checks internet access for package operations

    Runtime Configuration Example

    // Check if Python is installed at a specific path
string pythonPath = @"C:\Python39";
bool isInstalled = PythonRunTimeDiagnostics.IsPythonInstalled(pythonPath);

if (isInstalled)
{
    // Get comprehensive configuration
    var config = PythonRunTimeDiagnostics.GetPythonConfig(pythonPath);
    
    Console.WriteLine($"Python Configuration:");
    Console.WriteLine($"  Path: {config.RuntimePath}");
    Console.WriteLine($"  Version: {config.PythonVersion}");
    Console.WriteLine($"  Architecture: {config.BinType}");
    Console.WriteLine($"  DLL Path: {config.PythonDll}");
    Console.WriteLine($"  Package Path: {config.Packageinstallpath}");
    Console.WriteLine($"  Script Path: {config.ScriptPath}");
    
    // Check package manager availability
    PackageType packageType = PythonRunTimeDiagnostics.GetPackageType(pythonPath);
    Console.WriteLine($"  Package Manager: {packageType}");
    
    // Check if Conda is available
    string condaExe = PythonRunTimeDiagnostics.IsCondaInstalled(pythonPath);
    if (!string.IsNullOrEmpty(condaExe))
    {
        Console.WriteLine($"  Conda Available: {condaExe}");
    }
}

    Network and Package Validation

    // Check network connectivity for package operations
bool hasInternet = PythonRunTimeDiagnostics.CheckNet();
Console.WriteLine($"Internet connectivity: {hasInternet}");

if (hasInternet)
{
    // Check if PyPI is reachable
    bool pypiReachable = await PythonRunTimeDiagnostics.IsUrlReachableAsync("https://pypi.org");
    Console.WriteLine($"PyPI accessible: {pypiReachable}");
    
    // Check if a specific package exists
    var packageInfo = await PythonRunTimeDiagnostics.CheckIfPackageExistsAsync("numpy");
    if (packageInfo != null)
    {
        Console.WriteLine($"Package found: {packageInfo.PackageName} v{packageInfo.Version}");
        Console.WriteLine($"Description: {packageInfo.Description}");
    }
}

// Set up AI workspace folder
var dmeEditor = serviceProvider.GetService();
string aiFolder = PythonRunTimeDiagnostics.SetAiFolderPath(dmeEditor);
Console.WriteLine($"AI workspace configured at: {aiFolder}");

    Package Categorization Example

    // Initialize package category manager
var packageOperations = new PackageOperationManager(beepService, runtimeManager, environment, session);
var categoryManager = new PackageCategoryManager(beepService, packageOperations, progress);

// Auto-categorize all packages in environment
bool categorized = categoryManager.PopulateCommonPackageCategories(environment);
Console.WriteLine($"Automatically categorized packages: {categorized}");

// Get packages by category
var mlPackages = categoryManager.GetPackagesByCategory(environment, PackageCategory.MachineLearning);
var webPackages = categoryManager.GetPackagesByCategory(environment, PackageCategory.WebDevelopment);
var dataPackages = categoryManager.GetPackagesByCategory(environment, PackageCategory.DataScience);

Console.WriteLine($"Machine Learning packages: {mlPackages.Count}");
Console.WriteLine($"Web Development packages: {webPackages.Count}");
Console.WriteLine($"Data Science packages: {dataPackages.Count}");

// Manual category assignment
categoryManager.SetPackageCategory(environment, "custom-package", PackageCategory.Utilities);

// Bulk category updates
var categoryUpdates = new Dictionary
{
    ["flask"] = PackageCategory.WebDevelopment,
    ["numpy"] = PackageCategory.DataScience,
    ["tensorflow"] = PackageCategory.MachineLearning
};
categoryManager.UpdatePackageCategories(environment, categoryUpdates);

    Intelligent Category Suggestions

    // Get category suggestions for new packages
var packagesToInstall = new[] { "scikit-learn", "django", "matplotlib", "opencv-python" };
var suggestions = await categoryManager.SuggestCategoriesForPackagesAsync(packagesToInstall, environment);

Console.WriteLine("Category suggestions:");
foreach (var suggestion in suggestions)
{
    Console.WriteLine($"  {suggestion.Key} ? {suggestion.Value}");
}

// Apply suggestions
foreach (var suggestion in suggestions)
{
    categoryManager.SetPackageCategory(environment, suggestion.Key, suggestion.Value);
}

// Generate category report
var categoryCounts = new Dictionary();
foreach (PackageCategory category in Enum.GetValues())
{
    var packages = categoryManager.GetPackagesByCategory(environment, category);
    if (packages.Count > 0)
    {
        categoryCounts[category] = packages.Count;
    }
}

Console.WriteLine("\nPackage Distribution by Category:");
foreach (var kvp in categoryCounts.OrderByDescending(x => x.Value))
{
    Console.WriteLine($"  {kvp.Key}: {kvp.Value} packages");
}

    ? PackageOperationManager

    Handles core package operations including installation, uninstallation, and metadata retrieval.

    Namespace: Beep.Python.RuntimeEngine.PackageManagement

    Type: Class

    Purpose: Core package management operations and metadata handling

    Key Features

    • Multi-Manager Support: Works with both pip and conda
    • Async Operations: Non-blocking package operations
    • Progress Reporting: Real-time operation progress
    • Error Handling: Comprehensive error capture and reporting
    • Dependency Resolution: Intelligent dependency management

    Basic Package Operations

    // Initialize package operation manager
var packageOps = new PackageOperationManager(beepService, runtimeManager, environment, session);

// Install a single package
bool installed = await packageOps.InstallPackageAsync("numpy");
if (installed)
{
    Console.WriteLine("NumPy installed successfully");
}

// Install multiple packages
var packagesToInstall = new[] { "pandas", "matplotlib", "seaborn" };
var results = await packageOps.InstallPackagesAsync(packagesToInstall);

foreach (var result in results)
{
    string status = result.Value ? "?" : "?";
    Console.WriteLine($"{status} {result.Key}");
}

// Upgrade a package
bool upgraded = await packageOps.UpgradePackageAsync("numpy");
Console.WriteLine($"NumPy upgrade: {(upgraded ? "Success" : "Failed")}");

// Uninstall a package
bool uninstalled = await packageOps.UninstallPackageAsync("old-package");
Console.WriteLine($"Package removal: {(uninstalled ? "Success" : "Failed")}");

    Package Information and Validation

    // Check if package is installed
bool isInstalled = await packageOps.IsPackageInstalledAsync("numpy");
Console.WriteLine($"NumPy installed: {isInstalled}");

// Get package information
var packageInfo = await packageOps.GetPackageInfoAsync("pandas");
if (packageInfo != null)
{
    Console.WriteLine($"Package: {packageInfo.PackageName}");
    Console.WriteLine($"Version: {packageInfo.Version}");
    Console.WriteLine($"Status: {packageInfo.Status}");
    Console.WriteLine($"Category: {packageInfo.Category}");
}

// Get all installed packages
var installedPackages = await packageOps.GetInstalledPackagesAsync();
Console.WriteLine($"Total installed packages: {installedPackages.Count}");

// Check for package updates
var outdatedPackages = await packageOps.GetOutdatedPackagesAsync();
if (outdatedPackages.Any())
{
    Console.WriteLine("Packages with updates available:");
    foreach (var package in outdatedPackages)
    {
        Console.WriteLine($"  {package.PackageName}: {package.CurrentVersion} ? {package.LatestVersion}");
    }
}

// Validate package requirements
var requirements = new[] { "numpy>=1.20.0", "pandas>=1.3.0" };
bool requirementsMet = await packageOps.ValidateRequirementsAsync(requirements);
Console.WriteLine($"Requirements satisfied: {requirementsMet}");

    Package Set Operations

    // Initialize package set manager
var packageSetManager = new PackageSetManager(beepService, packageOperations);

// Get available package sets
var availableSets = packageSetManager.GetAvailablePackageSets();
Console.WriteLine("Available package sets:");
foreach (var set in availableSets)
{
    Console.WriteLine($"  {set.Name}: {set.Description}");
    Console.WriteLine($"    Category: {set.Category}");
    Console.WriteLine($"    Packages: {set.Packages.Count}");
}

// Install a complete package set
var dataScientistSet = packageSetManager.GetPackageSet("DataScientist");
if (dataScientistSet != null)
{
    Console.WriteLine($"Installing {dataScientistSet.Name} package set...");
    bool success = await packageSetManager.InstallPackageSetAsync(dataScientistSet);
    
    if (success)
    {
        Console.WriteLine("Data scientist package set installed successfully!");
        Console.WriteLine("Included packages:");
        foreach (var package in dataScientistSet.Packages)
        {
            Console.WriteLine($"  - {package.PackageName} {package.Version}");
        }
    }
}

// Create custom package set
var customSet = new PackageSet
{
    Name = "MyCustomSet",
    Description = "Custom package set for my project",
    Category = PackageCategory.Development,
    Packages = new List
    {
        new() { PackageName = "requests", Version = ">=2.25.0" },
        new() { PackageName = "click", Version = ">=8.0.0" },
        new() { PackageName = "pydantic", Version = ">=1.8.0" }
    }
};

// Register and install custom set
packageSetManager.RegisterPackageSet(customSet);
bool customInstalled = await packageSetManager.InstallPackageSetAsync(customSet);
Console.WriteLine($"Custom package set: {(customInstalled ? "Installed" : "Failed")}");

    ?? RequirementsFileManager

    Handles requirements.txt file generation, parsing, and environment synchronization.

    Namespace: Beep.Python.RuntimeEngine.PackageManagement

    Type: Class

    Purpose: Requirements file management and environment synchronization

    Requirements File Operations

    // Initialize requirements manager
var requirementsManager = new RequirementsFileManager(beepService, packageOperations);

// Generate requirements.txt from current environment
string requirementsPath = Path.Combine(environment.Path, "requirements.txt");
var options = new RequirementsGenerationOptions
{
    IncludeVersions = true,
    IncludeHashes = false,
    ExcludeDevelopmentPackages = true,
    SortAlphabetically = true
};

bool generated = await requirementsManager.GenerateRequirementsFileAsync(requirementsPath, options);
if (generated)
{
    Console.WriteLine($"Requirements file generated: {requirementsPath}");
    
    // Read and display contents
    var requirements = await File.ReadAllLinesAsync(requirementsPath);
    Console.WriteLine("Generated requirements:");
    foreach (string req in requirements)
    {
        Console.WriteLine($"  {req}");
    }
}

// Install from requirements file
bool installed = await requirementsManager.InstallFromRequirementsAsync(requirementsPath);
Console.WriteLine($"Requirements installation: {(installed ? "Success" : "Failed")}");

// Synchronize environment with requirements
var syncResult = await requirementsManager.SynchronizeWithRequirementsAsync(requirementsPath);
Console.WriteLine($"Packages to install: {syncResult.ToInstall.Count}");
Console.WriteLine($"Packages to upgrade: {syncResult.ToUpgrade.Count}");
Console.WriteLine($"Packages to remove: {syncResult.ToRemove.Count}");

if (syncResult.HasChanges)
{
    bool syncSuccess = await requirementsManager.ApplySynchronizationAsync(syncResult);
    Console.WriteLine($"Synchronization: {(syncSuccess ? "Completed" : "Failed")}");
}
